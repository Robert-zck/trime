017package com.osfans.trime;
018
019import java.util.List;
020
021import android.content.Context;
022import android.text.InputType;
023
024/** 管理多個{@link Keyboard 鍵盤} */
025public class KeyboardSwitch {
026
027  private final Context context;
028
029  private Keyboard[] mKeyboards;
030  private List<String> mKeyboardNames;
031  private int currentId, lastId;
032  private int currentDisplayWidth;
033
034  public KeyboardSwitch(Context context) {
035    this.context = context;
036    currentId = -1;
037    lastId = 0;
038    reset();
039  }
040
041  public void reset() {
042    mKeyboardNames = Config.get().getKeyboardNames();
043    int n = mKeyboardNames.size();
044    mKeyboards = new Keyboard[n];
045    for (int i = 0; i < n; i++ ) {
046      mKeyboards[i] = new Keyboard(context, mKeyboardNames.get(i));
047    }
048    setKeyboard(0);
049  }
050
051  public void setKeyboard(String name){
052    int i;
053    if (name == null || name.contentEquals(".default")) {
054      i = 0;
055    } else if (name.contentEquals(".prior")) { //前一個
056      i = currentId - 1;
057      if (i < 0) i = mKeyboards.length - 1;
058    } else {
059      i = mKeyboardNames.indexOf(name); //指定鍵盤
060    }
061    if (i < 0) i = currentId + 1; //默認下一個
062    setKeyboard(i);
063  }
064
065  public void setKeyboard(int i){
066    if (i < 0 || i >= mKeyboards.length) i = 0;
067    lastId = currentId;
068    currentId = i;
069  }
070
071  public void init(int displayWidth) {
072    if ((currentId >= 0) && (displayWidth == currentDisplayWidth)) {
073      return;
074    }
075
076    currentDisplayWidth = displayWidth;
077    reset();
078  }
079
080  public Keyboard getCurrentKeyboard() {
081    return mKeyboards[currentId];
082  }
083  
084  /**
085   * Switches to the appropriate keyboard based on the type of text being
086   * edited, for example, the symbol keyboard for numbers.
087   * 
088   * @param inputType one of the {@code InputType.TYPE_CLASS_*} values listed in
089   *     {@link android.text.InputType}.
090   */
091  public void onStartInput(int inputType) {
092    if ((inputType & InputType.TYPE_MASK_CLASS) == InputType.TYPE_CLASS_TEXT) {
093      int variation = inputType & InputType.TYPE_MASK_VARIATION;
094      if ((variation == InputType.TYPE_TEXT_VARIATION_EMAIL_ADDRESS)
095          || (variation == InputType.TYPE_TEXT_VARIATION_URI)
096          || (variation == InputType.TYPE_TEXT_VARIATION_PASSWORD)
097          || (variation == InputType.TYPE_TEXT_VARIATION_VISIBLE_PASSWORD)) {
098        currentId = 0;
099        setKeyboard(currentId);
100        mKeyboards[currentId].setShifted(false, mKeyboards[currentId].isShifted());
101      }
102    }
103  }
104
105  public boolean getAsciiMode() {
106    return getCurrentKeyboard().getAsciiMode();
107  }
108}
