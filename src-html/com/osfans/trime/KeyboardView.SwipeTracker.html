067    /** 處理按鍵、觸摸等輸入事件 */
068    public interface OnKeyboardActionListener {
069        
070        /**
071         * Called when the user presses a key. This is sent before the {@link #onKey} is called.
072         * For keys that repeat, this is only called once.
073         * @param primaryCode the unicode of the key being pressed. If the touch is not on a valid
074         * key, the value will be zero.
075         */
076        void onPress(int primaryCode);
077        
078        /**
079         * Called when the user releases a key. This is sent after the {@link #onKey} is called.
080         * For keys that repeat, this is only called once.
081         * @param primaryCode the code of the key that was released
082         */
083        void onRelease(int primaryCode);
084
085        void onEvent(Event event);
086
087        /**
088         * Send a key press to the listener.
089         * @param primaryCode this is the key that was pressed
090         * @param mask the codes for all the possible alternative keys
091         * with the primary code being the first. If the primary key code is
092         * a single character such as an alphabet or number or symbol, the alternatives
093         * will include other characters that may be on the same key or adjacent keys.
094         * These codes are useful to correct for accidental presses of a key adjacent to
095         * the intended key.
096         */
097        void onKey(int primaryCode, int mask);
098
099        /**
100         * Sends a sequence of characters to the listener.
101         * @param text the sequence of characters to be displayed.
102         */
103        void onText(CharSequence text);
104        
105        /**
106         * Called when the user quickly moves the finger from right to left.
107         */
108        void swipeLeft();
109        
110        /**
111         * Called when the user quickly moves the finger from left to right.
112         */
113        void swipeRight();
114        
115        /**
116         * Called when the user quickly moves the finger from up to down.
117         */
118        void swipeDown();
119        
120        /**
121         * Called when the user quickly moves the finger from down to up.
122         */
123        void swipeUp();
124    }

252        @Override
253        public void handleMessage(Message msg) {
254            switch (msg.what) {
255                case MSG_SHOW_PREVIEW:
256                    showKey(msg.arg1, msg.arg2);
257                    break;
258                case MSG_REMOVE_PREVIEW:
259                    mPreviewText.setVisibility(INVISIBLE);
260                    break;
261                case MSG_REPEAT:
262                    if (repeatKey()) {
263                        Message repeat = Message.obtain(this, MSG_REPEAT);
264                        sendMessageDelayed(repeat, REPEAT_INTERVAL);                        
265                    }
266                    break;
267                case MSG_LONGPRESS:
268                    openPopupIfRequired((MotionEvent) msg.obj);
269                    break;
270            }
271        }
272    };

371    private void initGestureDetector() {
372        mGestureDetector = new GestureDetector(getContext(), new GestureDetector.SimpleOnGestureListener() {
373            @Override
374            public boolean onFling(MotionEvent me1, MotionEvent me2, 
375                    float velocityX, float velocityY) {
376                if (mPossiblePoly) return false;
377                final float absX = Math.abs(velocityX);
378                final float absY = Math.abs(velocityY);
379                float deltaX = me2.getX() - me1.getX();
380                float deltaY = me2.getY() - me1.getY();
381                int travelX = 0; //getWidth() / 2; // Half the keyboard width
382                int travelY = 0; //getHeight() / 2; // Half the keyboard height
383                mSwipeTracker.computeCurrentVelocity(10);
384                final float endingVelocityX = mSwipeTracker.getXVelocity();
385                final float endingVelocityY = mSwipeTracker.getYVelocity();
386                boolean sendDownKey = false;
387                int type = 0;
388                if (velocityX > mSwipeThreshold && absY < absX && deltaX > travelX) {
389                    if (mDisambiguateSwipe && endingVelocityX < velocityX / 4) {
390                        sendDownKey = true;
391                        type = Key.SWIPE_RIGHT;
392                    } else {
393                        swipeRight();
394                        return true;
395                    }
396                } else if (velocityX < -mSwipeThreshold && absY < absX && deltaX < -travelX) {
397                    if (mDisambiguateSwipe && endingVelocityX > velocityX / 4) {
398                        sendDownKey = true;
399                        type = Key.SWIPE_LEFT;
400                    } else {
401                        swipeLeft();
402                        return true;
403                    }
404                } else if (velocityY < -mSwipeThreshold && absX < absY && deltaY < -travelY) {
405                    if (mDisambiguateSwipe && endingVelocityY > velocityY / 4) {
406                        sendDownKey = true;
407                        type = Key.SWIPE_UP;
408                    } else {
409                        swipeUp();
410                        return true;
411                    }
412                } else if (velocityY > mSwipeThreshold && absX < absY / 2 && deltaY > travelY) {
413                    if (mDisambiguateSwipe && endingVelocityY < velocityY / 4) {
414                        sendDownKey = true;
415                        type = Key.SWIPE_DOWN;
416                    } else {
417                        swipeDown();
418                        return true;
419                    }
420                }
421
422                if (sendDownKey) {
423                    showPreview(NOT_A_KEY);
424                    showPreview(mDownKey, type);
425                    detectAndSendKey(mDownKey, mStartX, mStartY, me1.getEventTime(), type);
426                    return true;
427                }
428                return false;
429            }
430        });
431
432        mGestureDetector.setIsLongpressEnabled(false);
433    }

447    /**
448     * Attaches a keyboard to this view. The keyboard can be switched at any time and the
449     * view will re-layout itself to accommodate the keyboard.
450     * @see Keyboard
451     * @see #getKeyboard()
452     * @param keyboard the keyboard to display in this view
453     */
454    public void setKeyboard(Keyboard keyboard) {
455        if (mKeyboard != null) {
456            showPreview(NOT_A_KEY);
457        }
458        // Remove any pending messages
459        removeMessages();
460        mKeyboard = keyboard;
461        List<Key> keys = mKeyboard.getKeys();
462        mKeys = keys.toArray(new Key[keys.size()]);
463        requestLayout();
464        // Hint to reallocate the buffer if the size changed
465        mKeyboardChanged = true;
466        invalidateAllKeys();
467        computeProximityThreshold(keyboard);
468        mMiniKeyboardCache.clear(); // Not really necessary to do every time, but will free up views
469        // Switching to a different keyboard should abort any pending keys so that the key up
470        // doesn't get delivered to the old or new keyboard
471        mAbortKey = true; // Until the next ACTION_DOWN
472    }

599    /**
600     * 計算水平和豎直方向的相鄰按鍵中心的平均距離的平方，這樣不需要做開方運算
601     * @param keyboard 鍵盤
602     */
603    private void computeProximityThreshold(Keyboard keyboard) {
604        if (keyboard == null) return;
605        final Key[] keys = mKeys;
606        if (keys == null) return;
607        int length = keys.length;
608        int dimensionSum = 0;
609        for (int i = 0; i < length; i++) {
610            Key key = keys[i];
611            dimensionSum += Math.min(key.width, key.height) + key.gap;
612        }
613        if (dimensionSum < 0 || length == 0) return;
614        mProximityThreshold = (int) (dimensionSum * 1.4f / length);
615        mProximityThreshold *= mProximityThreshold; // Square it
616    }

637    private void onBufferDraw() {
638        if (mBuffer == null || mKeyboardChanged) {
639            if (mBuffer == null || mKeyboardChanged &&
640                    (mBuffer.getWidth() != getWidth() || mBuffer.getHeight() != getHeight())) {
641                // Make sure our bitmap is at least 1x1
642                final int width = Math.max(1, getWidth());
643                final int height = Math.max(1, getHeight());
644                mBuffer = Bitmap.createBitmap(width, height, Bitmap.Config.ARGB_8888);
645                mCanvas = new Canvas(mBuffer);
646            }
647            invalidateAllKeys();
648            mKeyboardChanged = false;
649        }
650        final Canvas canvas = mCanvas;
651        canvas.clipRect(mDirtyRect, Op.REPLACE);
652        
653        if (mKeyboard == null) return;
654        
655        final Paint paint = mPaint;
656        Drawable keyBackground;
657        final Rect clipRegion = mClipRegion;
658        final Rect padding = mPadding;
659        final int kbdPaddingLeft = getPaddingLeft();
660        final int kbdPaddingTop = getPaddingTop();
661        final Key[] keys = mKeys;
662        final Key invalidKey = mInvalidatedKey;
663
664        boolean drawSingleKey = false;
665        if (invalidKey != null && canvas.getClipBounds(clipRegion)) {
666          // Is clipRegion completely contained within the invalidated key?
667          if (invalidKey.x + kbdPaddingLeft - 1 <= clipRegion.left &&
668                  invalidKey.y + kbdPaddingTop - 1 <= clipRegion.top &&
669                  invalidKey.x + invalidKey.width + kbdPaddingLeft + 1 >= clipRegion.right &&
670                  invalidKey.y + invalidKey.height + kbdPaddingTop + 1 >= clipRegion.bottom) {
671              drawSingleKey = true;
672          }
673        }
674        canvas.drawColor(0x00000000, PorterDuff.Mode.CLEAR);
675        final int keyCount = keys.length;
676        final float symbolTop = (mPaintSymbol.getTextSize());//mPaintSymbol.descent();
677        for (int i = 0; i < keyCount; i++) {
678            final Key key = keys[i];
679            if (drawSingleKey && invalidKey != key) {
680                continue;
681            }
682            int[] drawableState = key.getCurrentDrawableState();
683            keyBackground = new PaintDrawable(mKeyBackColor.getColorForState(drawableState, 0));
684            ((PaintDrawable)keyBackground).setCornerRadius(mRoundCorner);
685            mPaint.setColor(mKeyTextColor.getColorForState(drawableState, 0));
686            mPaintSymbol.setColor(key.pressed ? hilited_key_symbol_color: key_symbol_color);
687
688            // Switch the character to uppercase if shift is pressed
689            String label = key.getLabel();
690            String hint = key.hint;
691            int left = (key.width - padding.left - padding.right) / 2 + padding.left;
692            int top = padding.top;
693            
694            final Rect bounds = keyBackground.getBounds();
695            if (key.width != bounds.right || 
696                    key.height != bounds.bottom) {
697                keyBackground.setBounds(0, 0, key.width, key.height);
698            }
699            canvas.translate(key.x + kbdPaddingLeft, key.y + kbdPaddingTop);
700            keyBackground.draw(canvas);
701            
702            if (!label.isEmpty()) {
703                // For characters, use large font. For labels like "Done", use small font.
704                if (label.length() > 1) {
705                    paint.setTextSize(mLabelTextSize);
706                } else {
707                    paint.setTextSize(mKeyTextSize);
708                }
709                // Draw a drop shadow for the text
710                paint.setShadowLayer(mShadowRadius, 0, 0, mShadowColor);
711                // Draw the text
712                canvas.drawText(label, left,
713                    (key.height - padding.top - padding.bottom) / 2
714                            + (paint.getTextSize() - paint.descent()) / 2 + top,
715                    paint);
716
717                if (key.getLongClick() != null) {
718                    mPaintSymbol.setShadowLayer(mShadowRadius, 0, 0, mShadowColor);
719                    canvas.drawText(key.getSymbolLabel(), left, symbolTop + top, mPaintSymbol);
720                }
721
722                if (!hint.isEmpty()) {
723                    mPaintSymbol.setShadowLayer(mShadowRadius, 0, 0, mShadowColor);
724                    canvas.drawText(hint, left, key.height  - padding.bottom - symbolTop + top, mPaintSymbol);
725                }
726
727                // Turn off drop shadow
728                paint.setShadowLayer(0, 0, 0, 0);
729            }
730            canvas.translate(-key.x - kbdPaddingLeft, -key.y - kbdPaddingTop);
731        }
732        mInvalidatedKey = null;
733        // Overlay a dark rectangle to dim the keyboard
734        if (mMiniKeyboardOnScreen) {
735            paint.setColor((int) (mBackgroundDimAmount * 0xFF) << 24);
736            canvas.drawRect(0, 0, getWidth(), getHeight(), paint);
737        }
738
739        if (DEBUG && mShowTouchPoints) {
740            paint.setAlpha(128);
741            paint.setColor(0xFFFF0000);
742            canvas.drawCircle(mStartX, mStartY, 3, paint);
743            canvas.drawLine(mStartX, mStartY, mLastX, mLastY, paint);
744            paint.setColor(0xFF0000FF);
745            canvas.drawCircle(mLastX, mLastY, 3, paint);
746            paint.setColor(0xFF00FF00);
747            canvas.drawCircle((mStartX + mLastX) / 2, (mStartY + mLastY) / 2, 2, paint);
748        }
749        
750        mDrawPending = false;
751        mDirtyRect.setEmpty();
752    }

754    private int getKeyIndices(int x, int y, int[] allKeys) {
755        final Key[] keys = mKeys;
756        int primaryIndex = NOT_A_KEY;
757        int closestKey = NOT_A_KEY;
758        int closestKeyDist = mProximityThreshold + 1;
759        java.util.Arrays.fill(mDistances, Integer.MAX_VALUE);
760        int [] nearestKeyIndices = mKeyboard.getNearestKeys(x, y);
761        final int keyCount = nearestKeyIndices.length;
762        for (int i = 0; i < keyCount; i++) {
763            final Key key = keys[nearestKeyIndices[i]];
764            int dist = 0;
765            boolean isInside = key.isInside(x,y);
766            if (isInside) {
767                primaryIndex = nearestKeyIndices[i];
768            }
769
770            if (((mProximityCorrectOn 
771                    && (dist = key.squaredDistanceFrom(x, y)) < mProximityThreshold) 
772                    || isInside)
773                    && key.getCode() > 32) {
774                // Find insertion point
775                final int nCodes = 1;
776                if (dist < closestKeyDist) {
777                    closestKeyDist = dist;
778                    closestKey = nearestKeyIndices[i];
779                }
780                
781                if (allKeys == null) continue;
782                
783                for (int j = 0; j < mDistances.length; j++) {
784                    if (mDistances[j] > dist) {
785                        // Make space for nCodes codes
786                        System.arraycopy(mDistances, j, mDistances, j + nCodes,
787                                mDistances.length - j - nCodes);
788                        System.arraycopy(allKeys, j, allKeys, j + nCodes,
789                                allKeys.length - j - nCodes);
790                        allKeys[j] = key.getCode();
791                        mDistances[j] = dist;
792                        break;
793                    }
794                }
795            }
796        }
797        if (primaryIndex == NOT_A_KEY) {
798            primaryIndex = closestKey;
799        }
800        return primaryIndex;
801    }

827    private void showPreview(int keyIndex, int type) {
828        int oldKeyIndex = mCurrentKeyIndex;
829        final PopupWindow previewPopup = mPreviewPopup;
830        
831        mCurrentKeyIndex = keyIndex;
832        // Release the old key and press the new key
833        final Key[] keys = mKeys;
834        if (oldKeyIndex != mCurrentKeyIndex) {
835            if (oldKeyIndex != NOT_A_KEY && keys.length > oldKeyIndex) {
836                Key oldKey = keys[oldKeyIndex];
837                oldKey.onReleased(mCurrentKeyIndex == NOT_A_KEY);
838                invalidateKey(oldKeyIndex);
839            }
840            if (mCurrentKeyIndex != NOT_A_KEY && keys.length > mCurrentKeyIndex) {
841                Key newKey = keys[mCurrentKeyIndex];
842                newKey.onPressed();
843                invalidateKey(mCurrentKeyIndex);
844            }
845        }
846        // If key changed and preview is on ...
847        if (oldKeyIndex != mCurrentKeyIndex && mShowPreview) {
848            mHandler.removeMessages(MSG_SHOW_PREVIEW);
849            if (previewPopup.isShowing()) {
850                if (keyIndex == NOT_A_KEY) {
851                    mHandler.sendMessageDelayed(mHandler

853                            DELAY_AFTER_PREVIEW);
854                }
855            }
856            if (keyIndex != NOT_A_KEY) {
857                if (previewPopup.isShowing() && mPreviewText.getVisibility() == VISIBLE) {
858                    // Show right away, if it's already visible and finger is moving around
859                    showKey(keyIndex, type);
860                } else {
861                    mHandler.sendMessageDelayed(
862                            mHandler.obtainMessage(MSG_SHOW_PREVIEW, keyIndex, type),
863                            DELAY_BEFORE_PREVIEW);
864                }
865            }
866        }

874    private void showKey(final int keyIndex, int type) {
875        final PopupWindow previewPopup = mPreviewPopup;
876        final Key[] keys = mKeys;
877        if (keyIndex < 0 || keyIndex >= mKeys.length) return;
878        Key key = keys[keyIndex];
879        mPreviewText.setCompoundDrawables(null, null, null, null);
880        mPreviewText.setText(key.getPreviewText(type));
881        mPreviewText.measure(MeasureSpec.makeMeasureSpec(0, MeasureSpec.UNSPECIFIED), 
882                MeasureSpec.makeMeasureSpec(0, MeasureSpec.UNSPECIFIED));
883        int popupWidth = Math.max(mPreviewText.getMeasuredWidth(), key.width 
884                + mPreviewText.getPaddingLeft() + mPreviewText.getPaddingRight());
885        final int popupHeight = mPreviewHeight;
886        LayoutParams lp = mPreviewText.getLayoutParams();
887        if (lp != null) {
888            lp.width = popupWidth;
889            lp.height = popupHeight;
890        }
891        if (!mPreviewCentered) {
892            mPopupPreviewX = key.x - mPreviewText.getPaddingLeft() + getPaddingLeft();
893            mPopupPreviewY = key.y - popupHeight + mPreviewOffset;
894        } else {
895            // TODO: Fix this if centering is brought back
896            mPopupPreviewX = 160 - mPreviewText.getMeasuredWidth() / 2;
897            mPopupPreviewY = - mPreviewText.getMeasuredHeight();
898        }
899        mHandler.removeMessages(MSG_REMOVE_PREVIEW);
900        getLocationInWindow(mCoordinates);
901        mCoordinates[0] += mMiniKeyboardOffsetX; // Offset may be zero
902        mCoordinates[1] += mMiniKeyboardOffsetY; // Offset may be zero
903
904        // Set the preview background state
905        mPreviewText.getBackground().setState(
906                key.popupResId != 0 ? LONG_PRESSABLE_STATE_SET : EMPTY_STATE_SET);
907        mPopupPreviewX += mCoordinates[0];
908        mPopupPreviewY += mCoordinates[1];
909
910        // If the popup cannot be shown above the key, put it on the side
911        getLocationOnScreen(mCoordinates);
912        if (mPopupPreviewY + mCoordinates[1] < 0) {
913            // If the key you're pressing is on the left side of the keyboard, show the popup on
914            // the right, offset by enough to see at least one key to the left/right.
915            if (key.x + key.width <= getWidth() / 2) {
916                mPopupPreviewX += (int) (key.width * 2.5);
917            } else {
918                mPopupPreviewX -= (int) (key.width * 2.5);
919            }
920            mPopupPreviewY += popupHeight;
921        }
922
923        if (previewPopup.isShowing()) {
924            previewPopup.update(mPopupPreviewX, mPopupPreviewY,
925                    popupWidth, popupHeight);
926        } else {
927            previewPopup.setWidth(popupWidth);
928            previewPopup.setHeight(popupHeight);
929            previewPopup.showAtLocation(mPopupParent, Gravity.NO_GRAVITY, 
930                    mPopupPreviewX, mPopupPreviewY);
931        }
932        mPreviewText.setVisibility(VISIBLE);
933    }

1090              return true;
1091            }

1095            }
1096        }

1098    }

1122    @Override
1123    public boolean onTouchEvent(MotionEvent me) {
1124        // Convert multi-pointer up/down events to single up/down events to 
1125        // deal with the typical multi-pointer behavior of two-thumb typing
1126        final int pointerCount = me.getPointerCount();
1127        final int action = me.getAction();
1128        boolean result = false;
1129        final long now = me.getEventTime();
1130
1131        if (pointerCount != mOldPointerCount) {
1132            if (pointerCount == 1) {
1133                // Send a down event for the latest pointer
1134                MotionEvent down = MotionEvent.obtain(now, now, MotionEvent.ACTION_DOWN,
1135                        me.getX(), me.getY(), me.getMetaState());
1136                result = onModifiedTouchEvent(down, false);
1137                down.recycle();
1138                // If it's an up action, then deliver the up as well.
1139                if (action == MotionEvent.ACTION_UP) {
1140                    result = onModifiedTouchEvent(me, true);
1141                }
1142            } else {
1143                // Send an up event for the last pointer
1144                MotionEvent up = MotionEvent.obtain(now, now, MotionEvent.ACTION_UP,
1145                        mOldPointerX, mOldPointerY, me.getMetaState());
1146                result = onModifiedTouchEvent(up, true);
1147                up.recycle();
1148            }
1149        } else {
1150            if (pointerCount == 1) {
1151                result = onModifiedTouchEvent(me, false);
1152                mOldPointerX = me.getX();
1153                mOldPointerY = me.getY();
1154            } else {
1155                // Don't do anything when 2 pointers are down and moving.
1156                result = true;
1157            }
1158        }
1159        mOldPointerCount = pointerCount;
1160
1161        return result;
1162    }

1164    private boolean onModifiedTouchEvent(MotionEvent me, boolean possiblePoly) {
1165        int touchX = (int) me.getX() - getPaddingLeft();
1166        int touchY = (int) me.getY() - getPaddingTop();
1167        if (touchY >= -mVerticalCorrection)
1168            touchY += mVerticalCorrection;
1169        final int action = me.getAction();
1170        final long eventTime = me.getEventTime();
1171        int keyIndex = getKeyIndices(touchX, touchY, null);
1172        mPossiblePoly = possiblePoly;
1173
1174        // Track the last few movements to look for spurious swipes.
1175        if (action == MotionEvent.ACTION_DOWN) mSwipeTracker.clear();
1176        mSwipeTracker.addMovement(me);
1177
1178        // Ignore all motion events until a DOWN.
1179        if (mAbortKey
1180                && action != MotionEvent.ACTION_DOWN && action != MotionEvent.ACTION_CANCEL) {
1181            return true;
1182        }
1183
1184        if (mGestureDetector.onTouchEvent(me)) {
1185            showPreview(NOT_A_KEY);
1186            mHandler.removeMessages(MSG_REPEAT);
1187            mHandler.removeMessages(MSG_LONGPRESS);
1188            return true;
1189        }
1190        
1191        // Needs to be called after the gesture detector gets a turn, as it may have
1192        // displayed the mini keyboard
1193        if (mMiniKeyboardOnScreen && action != MotionEvent.ACTION_CANCEL) {
1194            return true;
1195        }
1196
1197        switch (action) {
1198            case MotionEvent.ACTION_DOWN:
1199                mAbortKey = false;
1200                mStartX = touchX;
1201                mStartY = touchY;
1202                mLastCodeX = touchX;
1203                mLastCodeY = touchY;
1204                mLastKeyTime = 0;
1205                mCurrentKeyTime = 0;
1206                mLastKey = NOT_A_KEY;
1207                mCurrentKey = keyIndex;
1208                mDownKey = keyIndex;
1209                mDownTime = me.getEventTime();
1210                mLastMoveTime = mDownTime;
1211                checkMultiTap(eventTime, keyIndex);
1212                mKeyboardActionListener.onPress(keyIndex != NOT_A_KEY ? 
1213                        mKeys[keyIndex].getCode() : 0);
1214                if (mCurrentKey >= 0 && mKeys[mCurrentKey].getClick().repeatable) {
1215                    mRepeatKeyIndex = mCurrentKey;
1216                    Message msg = mHandler.obtainMessage(MSG_REPEAT);
1217                    mHandler.sendMessageDelayed(msg, REPEAT_START_DELAY);
1218                    repeatKey();
1219                    // Delivering the key could have caused an abort
1220                    if (mAbortKey) {
1221                        mRepeatKeyIndex = NOT_A_KEY;
1222                        break;
1223                    }
1224                }
1225                if (mCurrentKey != NOT_A_KEY) {
1226                    Message msg = mHandler.obtainMessage(MSG_LONGPRESS, me);
1227                    mHandler.sendMessageDelayed(msg, LONGPRESS_TIMEOUT);
1228                }
1229                showPreview(keyIndex, 0);
1230                break;
1231
1232            case MotionEvent.ACTION_MOVE:
1233                boolean continueLongPress = false;
1234                if (keyIndex != NOT_A_KEY) {
1235                    if (mCurrentKey == NOT_A_KEY) {
1236                        mCurrentKey = keyIndex;
1237                        mCurrentKeyTime = eventTime - mDownTime;
1238                    } else {
1239                        if (keyIndex == mCurrentKey) {
1240                            mCurrentKeyTime += eventTime - mLastMoveTime;
1241                            continueLongPress = true;
1242                        } else if (mRepeatKeyIndex == NOT_A_KEY) {
1243                            resetMultiTap();
1244                            mLastKey = mCurrentKey;
1245                            mLastCodeX = mLastX;
1246                            mLastCodeY = mLastY;
1247                            mLastKeyTime =
1248                                    mCurrentKeyTime + eventTime - mLastMoveTime;
1249                            mCurrentKey = keyIndex;
1250                            mCurrentKeyTime = 0;
1251                        }
1252                    }
1253                }
1254                if (!continueLongPress) {
1255                    // Cancel old longpress
1256                    mHandler.removeMessages(MSG_LONGPRESS);
1257                    // Start new longpress if key has changed
1258                    if (keyIndex != NOT_A_KEY) {
1259                        Message msg = mHandler.obtainMessage(MSG_LONGPRESS, me);
1260                        mHandler.sendMessageDelayed(msg, LONGPRESS_TIMEOUT);
1261                    }
1262                }
1263                showPreview(mCurrentKey);
1264                mLastMoveTime = eventTime;
1265                break;
1266
1267            case MotionEvent.ACTION_UP:
1268                removeMessages();
1269                if (keyIndex == mCurrentKey) {
1270                    mCurrentKeyTime += eventTime - mLastMoveTime;
1271                } else {
1272                    resetMultiTap();
1273                    mLastKey = mCurrentKey;
1274                    mLastKeyTime = mCurrentKeyTime + eventTime - mLastMoveTime;
1275                    mCurrentKey = keyIndex;
1276                    mCurrentKeyTime = 0;
1277                }
1278                if (mCurrentKeyTime < mLastKeyTime && mCurrentKeyTime < DEBOUNCE_TIME
1279                        && mLastKey != NOT_A_KEY) {
1280                    mCurrentKey = mLastKey;
1281                    touchX = mLastCodeX;
1282                    touchY = mLastCodeY;
1283                }
1284                showPreview(NOT_A_KEY);
1285                Arrays.fill(mKeyIndices, NOT_A_KEY);
1286                // If we're not on a repeating key (which sends on a DOWN event)
1287                if (mRepeatKeyIndex == NOT_A_KEY && !mMiniKeyboardOnScreen && !mAbortKey) {
1288                    detectAndSendKey(mCurrentKey, touchX, touchY, eventTime);
1289                }
1290                invalidateKey(keyIndex);
1291                mRepeatKeyIndex = NOT_A_KEY;
1292                break;
1293            case MotionEvent.ACTION_CANCEL:
1294                removeMessages();
1295                dismissPopupKeyboard();
1296                mAbortKey = true;
1297                showPreview(NOT_A_KEY);
1298                invalidateKey(mCurrentKey);
1299                break;
1300        }
1301        mLastX = touchX;
1302        mLastY = touchY;
1303        return true;
1304    }

1383    /** 識別滑動手勢 */
1384    private static class SwipeTracker {
1385
1386        static final int NUM_PAST = 4;
1387        static final int LONGEST_PAST_TIME = 200;
1388
1389        final float mPastX[] = new float[NUM_PAST];
1390        final float mPastY[] = new float[NUM_PAST];
1391        final long mPastTime[] = new long[NUM_PAST];
1392
1393        float mYVelocity;
1394        float mXVelocity;
1395
1396        public void clear() {
1397            mPastTime[0] = 0;
1398        }
1399
1400        public void addMovement(MotionEvent ev) {
1401            long time = ev.getEventTime();
1402            final int N = ev.getHistorySize();
1403            for (int i=0; i<N; i++) {
1404                addPoint(ev.getHistoricalX(i), ev.getHistoricalY(i),
1405                        ev.getHistoricalEventTime(i));
1406            }
1407            addPoint(ev.getX(), ev.getY(), time);
1408        }
1409
1410        private void addPoint(float x, float y, long time) {
1411            int drop = -1;
1412            int i;
1413            final long[] pastTime = mPastTime;
1414            for (i=0; i<NUM_PAST; i++) {
1415                if (pastTime[i] == 0) {
1416                    break;
1417                } else if (pastTime[i] < time-LONGEST_PAST_TIME) {
1418                    drop = i;
1419                }
1420            }
1421            if (i == NUM_PAST && drop < 0) {
1422                drop = 0;
1423            }
1424            if (drop == i) drop--;
1425            final float[] pastX = mPastX;
1426            final float[] pastY = mPastY;
1427            if (drop >= 0) {
1428                final int start = drop+1;
1429                final int count = NUM_PAST-drop-1;
1430                System.arraycopy(pastX, start, pastX, 0, count);
1431                System.arraycopy(pastY, start, pastY, 0, count);
1432                System.arraycopy(pastTime, start, pastTime, 0, count);
1433                i -= (drop+1);
1434            }
1435            pastX[i] = x;
1436            pastY[i] = y;
1437            pastTime[i] = time;
1438            i++;
1439            if (i < NUM_PAST) {
1440                pastTime[i] = 0;
1441            }
1442        }
1443
1444        public void computeCurrentVelocity(int units) {
1445            computeCurrentVelocity(units, Float.MAX_VALUE);
1446        }
1447
1448        public void computeCurrentVelocity(int units, float maxVelocity) {
1449            final float[] pastX = mPastX;
1450            final float[] pastY = mPastY;
1451            final long[] pastTime = mPastTime;
1452
1453            final float oldestX = pastX[0];
1454            final float oldestY = pastY[0];
1455            final long oldestTime = pastTime[0];
1456            float accumX = 0;
1457            float accumY = 0;
1458            int N=0;
1459            while (N < NUM_PAST) {
1460                if (pastTime[N] == 0) {
1461                    break;
1462                }
1463                N++;
1464            }
1465
1466            for (int i=1; i < N; i++) {
1467                final int dur = (int)(pastTime[i] - oldestTime);
1468                if (dur == 0) continue;
1469                float dist = pastX[i] - oldestX;
1470                float vel = (dist/dur) * units;   // pixels/frame.
1471                if (accumX == 0) accumX = vel;
1472                else accumX = (accumX + vel) * .5f;
1473
1474                dist = pastY[i] - oldestY;
1475                vel = (dist/dur) * units;   // pixels/frame.
1476                if (accumY == 0) accumY = vel;
1477                else accumY = (accumY + vel) * .5f;
1478            }
1479            mXVelocity = accumX < 0.0f ? Math.max(accumX, -maxVelocity)
1480                    : Math.min(accumX, maxVelocity);
1481            mYVelocity = accumY < 0.0f ? Math.max(accumY, -maxVelocity)
1482                    : Math.min(accumY, maxVelocity);
1483        }
1484
1485        public float getXVelocity() {
1486            return mXVelocity;
1487        }
1488
1489        public float getYVelocity() {
1490            return mYVelocity;
1491        }
1492    }
