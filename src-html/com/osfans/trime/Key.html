062  public Key(Keyboard parent, Map<String,Object> mk) {
063    this(parent);
064    String s;
065    for (int i = 0; i < EVENT_NUM; i++) {
066      String eventType = eventTypes[i];
067      s = getString(mk, eventType);
068      if (!s.isEmpty()) events[i] = new Event(mKeyboard, s);
069    }
070    s = getString(mk, "composing");
071    if (!s.isEmpty()) {
072      composing = new Event(mKeyboard, s);
073    }
074    s = getString(mk, "has_menu");
075    if (!s.isEmpty()) {
076      has_menu = new Event(mKeyboard, s);
077    }
078    s = getString(mk, "paging");
079    if (!s.isEmpty()) {
080      paging = new Event(mKeyboard, s);
081    }
082    if (composing != null || has_menu != null || paging != null) mKeyboard.mComposingKeys.add(this);
083    s = getString(mk, "ascii");
084    if (!s.isEmpty()) ascii = new Event(mKeyboard, s);
085    label = getString(mk, "label");
086    hint = getString(mk, "hint");
087    if (isShift()) mKeyboard.mShiftKey = this;
088  }

126    /**
127     * Informs the key that it has been pressed, in case it needs to change its appearance or
128     * state.
129     * @see #onReleased(boolean)
130     */
131    public void onPressed() {
132        pressed = !pressed;
133    }
134    
135    /**
136     * Changes the pressed state of the key. If it is a sticky key, it will also change the
137     * toggled state of the key if the finger was release inside.
138     * @param inside whether the finger was released inside the key
139     * @see #onPressed()
140     */
141    public void onReleased(boolean inside) {
142        pressed = !pressed;
143        if (getClick().sticky) on = !on;
144    }
145
146    /**
147     * Detects if a point falls inside this key.
148     * @param x the x-coordinate of the point 
149     * @param y the y-coordinate of the point
150     * @return whether or not the point falls inside the key. If the key is attached to an edge,
151     * it will assume that all points between the key and the edge are considered to be inside
152     * the key.
153     */
154    public boolean isInside(int x, int y) {
155        boolean leftEdge = (edgeFlags & Keyboard.EDGE_LEFT) > 0;
156        boolean rightEdge = (edgeFlags & Keyboard.EDGE_RIGHT) > 0;
157        boolean topEdge = (edgeFlags & Keyboard.EDGE_TOP) > 0;
158        boolean bottomEdge = (edgeFlags & Keyboard.EDGE_BOTTOM) > 0;
159        if ((x >= this.x || (leftEdge && x <= this.x + this.width)) 
160                && (x < this.x + this.width || (rightEdge && x >= this.x)) 
161                && (y >= this.y || (topEdge && y <= this.y + this.height))
162                && (y < this.y + this.height || (bottomEdge && y >= this.y))) {
163            return true;
164        } else {
165            return false;
166        }
167    }
168
169    /**
170     * Returns the square of the distance between the center of the key and the given point.
171     * @param x the x-coordinate of the point
172     * @param y the y-coordinate of the point
173     * @return the square of the distance of the point from the center of the key
174     */
175    public int squaredDistanceFrom(int x, int y) {
176        int xDist = this.x + width / 2 - x;
177        int yDist = this.y + height / 2 - y;
178        return xDist * xDist + yDist * yDist;
179    }
180    
181    /**
182     * Returns the drawable state for the key, based on the current state and type of the key.
183     * @return the drawable state of the key.
184     * @see android.graphics.drawable.StateListDrawable#setState(int[])
185     */
186    public int[] getCurrentDrawableState() {
187        int[] states = KEY_STATE_NORMAL;
188
189        if (on) {
190            if (pressed) {
191                states = KEY_STATE_PRESSED_ON;
192            } else {
193                states = KEY_STATE_NORMAL_ON;
194            }
195        } else {
196            if (getClick().sticky || getClick().functional) {
197                if (pressed) {
198                    states = KEY_STATE_PRESSED_OFF;
199                } else {
200                    states = KEY_STATE_NORMAL_OFF;
201                }
202            } else {
203                if (pressed) {
204                    states = KEY_STATE_PRESSED;
205                }
206            }
207        }
208        return states;
209    }

211  public boolean isShift() {
212    int c = getCode();
213    return (c == KeyEvent.KEYCODE_SHIFT_LEFT || c == KeyEvent.KEYCODE_SHIFT_RIGHT);
214  }

216  public static String getString(Map m, String k) {
217    if (m.containsKey(k)) return m.get(k).toString();
218    return "";
219  }
220
221  public static Object getValue(Map m, String k, Object o) {
222    return m.containsKey(k) ? m.get(k) : o;
223  }
224
225  public static double getDouble(Map m, String k, Object i) {
226    Object o = getValue(m, k, i);
227    if (o instanceof Integer) return ((Integer)o).doubleValue();
228    else if (o instanceof Float) return ((Float)o).doubleValue();
229    else if (o instanceof Double) return ((Double)o).doubleValue();
230    return 0f;
231  }
232
233  public Event getEvent() {
234    if (ascii != null && Rime.isAsciiMode()) return ascii;
235    if (paging != null && Rime.isPaging()) return paging;
236    if (has_menu != null && Rime.hasMenu()) return has_menu;
237    if (composing != null && Rime.isComposing()) return composing;
238    return getClick();
239  }
240
241  public Event getClick() {
242    return events[CLICK];
243  }
244
245  public Event getLongClick() {
246    return events[LONG_CLICK];
247  }
248
249  public Event getEvent(int i) {
250    Event e = null;
251    if (i >= 0 && i <= EVENT_NUM) e = events[i];
252    if (e != null) return e;
253    return getClick();
254  }
255
256  public int getCode() {
257    return getEvent(CLICK).code;
258  }
259
260  public int getCode(int type) {
261    return getEvent(type).code;
262  }
263
264  public String getLabel() {
265    Event event = getEvent();
266    if (!label.isEmpty() && event == getClick()
267    && (ascii == null && !Rime.isAsciiMode()))
268      return label; //中文狀態顯示標籤
269    return event.getLabel();
270  }
271
272  public String getPreviewText(int type) {
273    if (type == CLICK) return getEvent().getPreviewText();
274    return getEvent(type).getPreviewText();
275  }
