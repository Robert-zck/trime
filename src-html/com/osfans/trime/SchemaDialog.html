017package com.osfans.trime;
018
019import android.app.AlertDialog;
020import android.content.DialogInterface;
021import android.content.Context;
022import android.util.Log;
023import android.widget.Toast;
024
025import java.util.Arrays;
026import java.util.ArrayList;
027import java.util.List;
028import java.util.Map;
029import java.util.HashMap;
030import java.util.Collections;
031import java.util.Comparator;
032
033public class SchemaDialog {
034  boolean[] checkedSchemaItems;
035  String[] schemaItems;
036  AlertDialog dialog;
037
038  public class SortByName implements Comparator<Map<String,String>>{
039    public int compare(Map<String,String> m1, Map<String,String> m2) {
040      String s1 = m1.get("schema_id");
041      String s2 = m2.get("schema_id");
042      return s1.compareTo(s2);
043    }
044  }
045
046  private void selectSchema() {
047    List<String> checkedIds = new ArrayList<String>();
048    int i = 0;
049    for (boolean b: checkedSchemaItems) {
050      if (b) checkedIds.add(schemaItems[i]);
051      i++;
052    }
053    int n = checkedIds.size();
054    if (n > 0) {
055      String[] schema_id_list = new String[n];
056      checkedIds.toArray(schema_id_list);
057      Rime.select_schemas(schema_id_list);
058      Pref.deploy();
059    }
060  }
061
062  public SchemaDialog(Context context) {
063    List<Map<String,String>> schemas = Rime.get_available_schema_list();
064    if (schemas == null || schemas.size() == 0) {
065      Toast.makeText(context, R.string.no_schemas, Toast.LENGTH_LONG).show();
066      return;
067    }
068    Collections.sort(schemas, new SortByName());
069    List<Map<String,String>> selected_schemas = Rime.get_selected_schema_list();
070    List<String> selected_Ids = new ArrayList<String>();
071    int n = schemas.size();
072    String[] schemaNames = new String[n];
073    String schema_id;
074    checkedSchemaItems = new boolean[n];
075    schemaItems = new String[n];
076    int i = 0;
077    if (selected_schemas.size() > 0) {
078      for (Map<String,String> m: selected_schemas) {
079        selected_Ids.add(m.get("schema_id"));
080      }
081    }
082    for (Map<String,String> m: schemas) {
083      schemaNames[i] = m.get("name");
084      schema_id = m.get("schema_id");
085      schemaItems[i] = schema_id;
086      checkedSchemaItems[i] = selected_Ids.contains(schema_id);
087      i++;
088    }
089    dialog = new AlertDialog.Builder(context)
090      .setTitle(R.string.pref_schemas)
091      .setCancelable(true)
092      .setNegativeButton(android.R.string.cancel, null)
093      .setPositiveButton(android.R.string.ok, new DialogInterface.OnClickListener() {
094        public void onClick(DialogInterface di, int id) {
095          selectSchema();
096        }
097      })
098      .setMultiChoiceItems(schemaNames, checkedSchemaItems, new DialogInterface.OnMultiChoiceClickListener() {
099        public void onClick(DialogInterface di, int id, boolean isChecked) {
100          checkedSchemaItems[id] = isChecked;
101        }
102      })
103      .create();
104  }
105
106  public AlertDialog getDialog() {
107    return dialog;
108  }
109
110  public void show() {
111    if (dialog != null) dialog.show();
112  }
113}
