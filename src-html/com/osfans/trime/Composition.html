017package com.osfans.trime;
018
019import android.content.Context;
020import android.util.AttributeSet;
021import android.view.View;
022import android.view.MotionEvent;
023import android.widget.TextView;
024import android.widget.TextView.BufferType;
025import android.util.Log;
026import android.text.SpannableString;
027import android.text.style.ForegroundColorSpan;
028import android.text.style.BackgroundColorSpan;
029import android.text.style.UnderlineSpan;
030
031/** 編碼區，顯示已輸入的按鍵編碼，可使用方向鍵或觸屏移動光標位置 */
032public class Composition extends TextView {
033  private boolean soft_cursor;
034  private String soft_cursor_text;
035  private static String caret = "‸";
036  private ForegroundColorSpan hilited_text_color_span;
037  private BackgroundColorSpan hilited_back_color_span;
038  private UnderlineSpan underline_span = new UnderlineSpan();
039
040  public Composition(Context context, AttributeSet attrs) {
041    super(context, attrs);
042    reset();
043  }
044
045  public boolean onTouchEvent(MotionEvent event) {
046    int n = getOffsetForPosition(event.getX(),event.getY());
047    if (event.getAction() == MotionEvent.ACTION_UP && n >= 0) {
048      String s = getText().toString().substring(0, n);
049      if (soft_cursor) s = s.replace(soft_cursor_text, "").replace(" ", "");
050      n = s.length();
051      Rime.RimeSetCaretPos(n);
052      Trime.getService().updateComposing();
053    }
054    return true;
055  }
056
057  public void reset() {
058    Config config = Config.get();
059    setBackgroundColor(config.getColor("text_back_color"));
060    setTextColor(config.getColor("text_color"));
061    setTextSize(config.getInt("text_size"));
062    hilited_text_color_span = new ForegroundColorSpan(config.getColor("hilited_text_color"));
063    hilited_back_color_span = new BackgroundColorSpan(config.getColor("hilited_back_color"));
064    //setHighlightColor(config.getColor("hilited_back_color"));
065    setHeight(config.getPixel("text_height"));
066    setTypeface(config.getFont("text_font"));
067    boolean show = config.getBoolean("show_text");
068    setVisibility(show ? View.VISIBLE : View.GONE);
069    soft_cursor = config.getBoolean("soft_cursor");
070    soft_cursor_text = config.getString("soft_cursor_text");
071    if (soft_cursor_text == null || soft_cursor_text.isEmpty()) soft_cursor_text = caret;
072  }
073
074  public void setText() {
075    if (getVisibility() != View.VISIBLE) return;
076    Rime.RimeComposition r = Rime.getComposition();
077    String s = "";
078    if (r != null) s = r.getText();
079    if (soft_cursor) s = s.replace(caret, soft_cursor_text);
080    setText(s, BufferType.SPANNABLE);
081    int n = s.length();
082    if (n > 0 && r != null) {
083      SpannableString ss = (SpannableString)getText();
084      int start = r.getStart();
085      int end = r.getEnd();
086      ss.setSpan(hilited_text_color_span, start, end, SpannableString.SPAN_INCLUSIVE_INCLUSIVE);
087      ss.setSpan(hilited_back_color_span, start, end, SpannableString.SPAN_INCLUSIVE_INCLUSIVE);
088      ss.setSpan(underline_span, start, end, SpannableString.SPAN_INCLUSIVE_INCLUSIVE);
089    }
090  }
091}
