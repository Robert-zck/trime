001package com.osfans.trime;
002
003import java.util.ArrayList;
004
005import android.speech.RecognitionListener;
006import android.speech.RecognizerIntent;
007import android.speech.SpeechRecognizer;
008import android.content.Context;
009import android.content.Intent;
010import android.os.Bundle;
011import android.util.Log;
012import android.widget.Toast;
013
014public class Speech implements RecognitionListener {
015  private SpeechRecognizer speech = null;
016  private Intent recognizerIntent;
017  private String TAG = "Speech";
018  private Context context;
019
020  public Speech(Context context) {
021    this.context = context;
022    speech = SpeechRecognizer.createSpeechRecognizer(context);
023    speech.setRecognitionListener(this);
024    recognizerIntent = new Intent(RecognizerIntent.ACTION_RECOGNIZE_SPEECH);
025    recognizerIntent.putExtra(RecognizerIntent.EXTRA_MAX_RESULTS, 1);
026    //recognizerIntent.putExtra(RecognizerIntent.EXTRA_LANGUAGE_PREFERENCE, "en");
027    //recognizerIntent.putExtra(RecognizerIntent.EXTRA_CALLING_PACKAGE, context.getPackageName());
028    //recognizerIntent.putExtra(RecognizerIntent.EXTRA_LANGUAGE_MODEL, RecognizerIntent.LANGUAGE_MODEL_FREE_FORM);
029    //recognizerIntent.putExtra(RecognizerIntent.EXTRA_PROMPT, "開始語音");
030  }
031
032  public void alert(String text) {
033    Toast.makeText(context, text, Toast.LENGTH_SHORT).show();
034  }
035
036  public void start() {
037    speech.startListening(recognizerIntent);
038  }
039
040  public void stop() {
041    speech.stopListening();
042  }
043
044  public void destory() {
045    if (speech != null) {
046      speech.destroy();
047    }
048  }
049
050  @Override
051  public void onBeginningOfSpeech() {
052    Log.i(TAG, "onBeginningOfSpeech");
053  }
054
055  @Override
056  public void onBufferReceived(byte[] buffer) {
057    Log.i(TAG, "onBufferReceived: " + buffer);
058  }
059
060  @Override
061  public void onEndOfSpeech() {
062    Log.i(TAG, "onEndOfSpeech");
063  }
064
065  @Override
066  public void onError(int errorCode) {
067    speech.stopListening();
068    speech.destroy();
069    String errorMessage = getErrorText(errorCode);
070    alert(errorMessage);
071  }
072
073  @Override
074  public void onEvent(int arg0, Bundle arg1) {
075    Log.i(TAG, "onEvent");
076  }
077
078  @Override
079  public void onPartialResults(Bundle arg0) {
080    Log.i(TAG, "onPartialResults");
081  }
082
083  @Override
084  public void onReadyForSpeech(Bundle arg0) {
085    Log.i(TAG, "onReadyForSpeech");
086    alert("請開始說話：");
087  }
088
089  @Override
090  public void onResults(Bundle results) {
091    stop();
092    destory();
093    Log.i(TAG, "onResults");
094    Trime trime = Trime.getService();
095    if (trime == null) return;
096    ArrayList<String> matches = results.getStringArrayList(SpeechRecognizer.RESULTS_RECOGNITION);
097    String opencc_config = Config.get(context).getString("speech_opencc_config");
098    for (String result : matches) trime.commitText(Rime.openccConvert(result, opencc_config));
099  }
100
101  @Override
102  public void onRmsChanged(float rmsdB) {
103    Log.i(TAG, "onRmsChanged: " + rmsdB);
104  }
105
106  public static String getErrorText(int errorCode) {
107    String message;
108    switch (errorCode) {
109    case SpeechRecognizer.ERROR_AUDIO:
110    message = "錄音錯誤";
111    break;
112    case SpeechRecognizer.ERROR_CLIENT:
113    message = "客戶端錯誤";
114    break;
115    case SpeechRecognizer.ERROR_INSUFFICIENT_PERMISSIONS:
116    message = "權限不足";
117    break;
118    case SpeechRecognizer.ERROR_NETWORK:
119    message = "網絡錯誤";
120    break;
121    case SpeechRecognizer.ERROR_NETWORK_TIMEOUT:
122    message = "網絡超時";
123    break;
124    case SpeechRecognizer.ERROR_NO_MATCH:
125    message = "未能識別";
126    break;
127    case SpeechRecognizer.ERROR_RECOGNIZER_BUSY:
128    message = "識別服務忙";
129    break;
130    case SpeechRecognizer.ERROR_SERVER:
131    message = "服務器錯誤";
132    break;
133    case SpeechRecognizer.ERROR_SPEECH_TIMEOUT:
134    message = "無語音輸入";
135    break;
136    default:
137    message = "未知錯誤";
138    break;
139    }
140    return message;
141  }
142}
