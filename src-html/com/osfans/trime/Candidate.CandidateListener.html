131    updateCandidateWidth();
132    if (getCandNum() > 0) {

137  /**
138   * Picks the highlighted candidate.
139   *
140   * @return {@code false} if no candidate is highlighted and picked.
141   */
142  public boolean pickHighlighted(int index) {
143    if ((highlightIndex != -1) && (listener != null)) {
144      listener.onPickCandidate(index == -1 ? highlightIndex : index);
145      return true;
146    }
147    return false;
148  }
149
150  private boolean updateHighlight(int x, int y) {
151    int index = getCandidateIndex(x, y);
152    if (index != -1) {
153      highlightIndex = index;
154      invalidate();
155      return true;
156    }
157    return false;
158  }
159
160  private void removeHighlight() {
161    highlightIndex = -1;
162    invalidate();
163    requestLayout();
164  }
165
166  private void drawHighlight(Canvas canvas) {
167    if (highlightIndex >= 0) {
168      candidateHighlight.setBounds(candidateRect[highlightIndex]);
169      candidateHighlight.draw(canvas);
170    }
171  }

173  private void drawText(String s, Canvas canvas, Paint paint, Typeface font, float center, float y) {
174    if (s == null) return;
175    int length = s.length();
176    if (length == 0) return;
177    int points = s.codePointCount(0, length);
178    float x = center - paint.measureText(s) / 2;
179    if (tfHanB != null && length > points) {
180      for (int offset = 0; offset < length; ) {
181        int codepoint = s.codePointAt(offset);
182        int charCount = Character.charCount(codepoint);
183        int end = offset + charCount;
184        paint.setTypeface(Character.isSupplementaryCodePoint(codepoint) ? tfHanB : font);
185        canvas.drawText(s, offset, end, x, y, paint);
186        x += paint.measureText(s, offset, end);
187        offset += charCount;
188      }
189    } else {
190      paint.setTypeface(font);
191      canvas.drawText(s, x, y, paint);
192    }
193  }
194
195  private void drawCandidates(Canvas canvas) {
196    if (candidates == null) return;
197
198    float x = 0;
199    float y = 0;
200    int i = 0;
201    float comment_x, comment_y;
202    float comment_width;
203    String candidate, comment;
204
205    y = candidateRect[0].centerY() - (paintCandidate.ascent() + paintCandidate.descent()) / 2;
206    if (show_comment && comment_on_top) y += comment_height / 2;
207    comment_y = comment_height / 2 - (paintComment.ascent() + paintComment.descent()) / 2;
208    if (show_comment && !comment_on_top) comment_y += candidateRect[0].bottom - comment_height;
209
210    while (i < num_candidates) {
211      // Calculate a position where the text could be centered in the rectangle.
212      x = candidateRect[i].centerX();
213      if (show_comment) {
214        comment = getComment(i);
215        if (comment != null && !comment.isEmpty()) {
216          comment_width = paintComment.measureText(comment);
217          if (comment_on_top) {
218            comment_x = candidateRect[i].centerX();
219          } else {
220            x -= comment_width / 2;
221            comment_x = candidateRect[i].right -  comment_width / 2;
222          }
223          paintComment.setColor(highlightIndex == i ? hilited_comment_text_color : comment_text_color);
224          drawText(comment, canvas, paintComment, tfComment, comment_x, comment_y);
225        }
226      }
227      paintCandidate.setColor(highlightIndex == i ? hilited_candidate_text_color : candidate_text_color);
228      drawText(getCandidate(i), canvas, paintCandidate, tfCandidate, x, y);
229      // Draw the separator at the right edge of each candidate.
230      candidateSeparator.setBounds(
231        candidateRect[i].right - candidateSeparator.getIntrinsicWidth(),
232        candidateRect[i].top,
233        candidateRect[i].right,
234        candidateRect[i].bottom);
235      candidateSeparator.draw(canvas);
236      i++;
237    }
238    for (int j = -4; j >= -5; j--) { // -4: left, -5: right
239      candidate = getCandidate(j);
240      if (candidate == null) continue;
241      paintSymbol.setColor(highlightIndex == i ? hilited_comment_text_color : comment_text_color);
242      x = candidateRect[i].centerX() - paintSymbol.measureText(candidate) / 2;
243      canvas.drawText(candidate, x, y, paintSymbol);
244      candidateSeparator.setBounds(
245        candidateRect[i].right - candidateSeparator.getIntrinsicWidth(),
246        candidateRect[i].top,
247        candidateRect[i].right,
248        candidateRect[i].bottom);
249      candidateSeparator.draw(canvas);
250      i++;
251    }
252  }

254  @Override
255  protected void onDraw(Canvas canvas) {
256    if (canvas == null) {
257      return;
258    }
259    super.onDraw(canvas);
260
261    drawHighlight(canvas);
262    drawCandidates(canvas);
263  }

265  private void updateCandidateWidth() {
266    final int top = 0;
267    final int bottom = getHeight();
268    int i = 0;
269    int x = 0;
270    if (Rime.hasLeft()) x += getCandidateWidth(-4);
271    getCandNum();
272    for (i = 0; i < num_candidates; i++) candidateRect[i] = new Rect(x, top, x += getCandidateWidth(i), bottom);
273    if (Rime.hasLeft()) candidateRect[i++] = new Rect(0, top, (int)getCandidateWidth(-4), bottom);
274    if (Rime.hasRight()) candidateRect[i++] = new Rect(x, top, x += getCandidateWidth(-5), bottom);
275    LayoutParams params = getLayoutParams();
276    params.width = x;
277    params.height = candidate_view_height;
278    if (show_comment && comment_on_top) params.height += comment_height;
279    setLayoutParams(params);
280  }

282  @Override
283  protected void onSizeChanged(int w, int h, int oldw, int oldh) {
284    super.onSizeChanged(w, h, oldw, oldh);
285    updateCandidateWidth();
286  }

288  @Override
289  public boolean onTouchEvent(MotionEvent me) {
290    int action = me.getAction();
291    int x = (int) me.getX();
292    int y = (int) me.getY();
293
294    switch (action) {
295      case MotionEvent.ACTION_DOWN:
296      case MotionEvent.ACTION_MOVE:
297        updateHighlight(x, y);
298        break;
299      case MotionEvent.ACTION_UP:
300        if (updateHighlight(x, y)) {
301          pickHighlighted(-1);
302        }
303        break;
304    }
305    return true;
306  }

308  /**
309   * Returns the index of the candidate which the given coordinate points to.
310   * 
311   * @return -1 if no candidate is mapped to the given (x, y) coordinate.
312   */
313  private int getCandidateIndex(int x, int y) {
314    Rect r = new Rect();
315
316    int j = 0;
317    for (int i = 0; i < num_candidates; i++) {
318      // Enlarge the rectangle to be more responsive to user clicks.
319      r.set(candidateRect[j++]);
320      r.inset(0, CANDIDATE_TOUCH_OFFSET);
321      if (r.contains(x, y)) {
322        // Returns -1 if there is no candidate in the hitting rectangle.
323        return (i < num_candidates) ? i : -1;
324      }
325    }
326
327    if (Rime.hasLeft()) { //Page Up
328      r.set(candidateRect[j++]);
329      r.inset(0, CANDIDATE_TOUCH_OFFSET);
330      if (r.contains(x, y)) {
331        return -4;
332      }
333    }
334
335    if (Rime.hasRight()) { //Page Down
336      r.set(candidateRect[j++]);
337      r.inset(0, CANDIDATE_TOUCH_OFFSET);
338      if (r.contains(x, y)) {
339        return -5;
340      }
341    }
342
343    return -1;
344  }

346  private int getCandNum() {
347    candidates = Rime.getCandidates();
348    highlightIndex = Rime.getCandHighlightIndex();
349    num_candidates = candidates == null ? 0 : candidates.length;
350    return num_candidates;
351  }
352
353  private String getCandidate(int i) {
354    String s = null;
355    if (candidates != null && i >= 0) s = candidates[i].text;
356    else if (i == -4 && Rime.hasLeft()) s = "◀";
357    else if (i == -5 && Rime.hasRight()) s = "▶";
358    return s;
359  }
360
361  private String getComment(int i) {
362    String s = null;
363    if (candidates != null && i >= 0) s = candidates[i].comment;
364    return s;
365  }
366
367  private float getCandidateWidth(int i) {
368    String s = getCandidate(i);
369    float n = (s == null ? 0 : s.codePointCount(0, s.length()));
370    n += n < 2 ? 0.8f : 0.4f;
371    float x = n * candidate_text_size;
372    if (i >= 0 && show_comment) {
373      String comment = getComment(i);
374      if (comment != null) {
375        float x2 = paintComment.measureText(comment);
376        if (comment_on_top) { if (x2 > x) x = x2; } //提示在上方
377        else x += x2;  //提示在右方
378      }
379    }
380    return x;
381  }
