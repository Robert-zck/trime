017package com.osfans.trime;
018
019import android.content.Context;
020import android.media.AudioManager;
021import android.os.Vibrator;
022import android.view.KeyEvent;
023import android.speech.tts.TextToSpeech;
024
025import java.util.Locale;
026
027/** 處理按鍵聲音、震動、朗讀等效果 */
028public class Effect {
029  private int duration = 30;
030  private float volume = -1.0f;
031
032  private final Context context;
033
034  private boolean vibrateOn;
035  private Vibrator vibrator;
036  private boolean soundOn;
037  private AudioManager audioManager;
038  private boolean isSpeakCommit, isSpeakKey;
039  private TextToSpeech mTTS;
040
041  public Effect(Context context) {
042    this.context = context;
043  }
044
045  public void reset() {
046    Config config = Config.get();
047    vibrateOn = config.getBoolean("key_vibrate");
048    duration = config.getInt("key_vibrate_duration");
049    if (vibrateOn && (vibrator == null)) {
050      vibrator =
051        (Vibrator) context.getSystemService(Context.VIBRATOR_SERVICE);
052    }
053
054    soundOn = config.getBoolean("key_sound");
055    volume = config.getFloat("key_sound_volume");
056    if (soundOn && (audioManager == null)) {
057      audioManager = 
058        (AudioManager) context.getSystemService(Context.AUDIO_SERVICE);
059    }
060
061    isSpeakCommit = config.getBoolean("speak_commit");
062    isSpeakKey = config.getBoolean("speak_key");
063    if (mTTS == null && (isSpeakCommit || isSpeakKey)) {
064      mTTS = new TextToSpeech(context, new TextToSpeech.OnInitListener() {
065        public void onInit(int status) {
066          //初始化結果
067        }
068      });
069    }
070  }
071
072  public void vibrate() {
073    if (vibrateOn && (vibrator != null)) vibrator.vibrate(duration);
074  }
075
076  public void playSound(final int code) {
077    if (soundOn && (audioManager != null)) {
078      final int sound;
079      switch (code) {
080        case KeyEvent.KEYCODE_DEL:
081            sound = AudioManager.FX_KEYPRESS_DELETE;
082            break;
083        case KeyEvent.KEYCODE_ENTER:
084            sound = AudioManager.FX_KEYPRESS_RETURN;
085            break;
086        case KeyEvent.KEYCODE_SPACE:
087            sound = AudioManager.FX_KEYPRESS_SPACEBAR;
088            break;
089        default:
090            sound = AudioManager.FX_KEYPRESS_STANDARD;
091            break;
092      }
093      audioManager.playSoundEffect(sound, volume);
094    }
095  }
096
097  public void setLanguage(Locale loc) {
098    if (mTTS != null) mTTS.setLanguage(loc);
099  }
100
101  public void speak(CharSequence text) {
102    if (text!= null && mTTS != null) mTTS.speak(text.toString(), TextToSpeech.QUEUE_FLUSH, null);
103  }
104
105  public void speakCommit(CharSequence text) {
106    if (isSpeakCommit) speak(text);
107  }
108
109  public void speakKey(CharSequence text) {
110    if (isSpeakKey) speak(text);
111  }
112
113  public void speakKey(int code) {
114    if (code <= 0) return;
115    String text = KeyEvent.keyCodeToString(code).replace("KEYCODE_","").replace("_", " ").toLowerCase();
116    speakKey(text);
117  }
118
119  public void destory() {
120     if (mTTS != null) {
121       mTTS.stop();
122       mTTS.shutdown();
123       mTTS = null;
124     }
125  }
126}
