017package com.osfans.trime;
018
019import android.view.KeyEvent;
020import android.util.Log;
021
022import java.io.IOException;
023import java.util.ArrayList;
024import java.util.Arrays;
025import java.util.List;
026import java.util.Map;
027import java.util.HashMap;
028
029/** {@link Key 按鍵}的各種事件（單擊、長按、滑動等） */
030public class Event {
031  private String TAG = "Event";
032  private Keyboard mKeyboard;
033  int code, mask;
034  String text, label, preview;
035  List<String> states;
036  String command, option, select, toggle;
037  public boolean functional, repeatable, sticky;
038
039  public Event(Keyboard keyboard, String s) {
040    mKeyboard = keyboard;
041    if (Key.presetKeys.containsKey(s)) {
042      Map m = Key.presetKeys.get(s);
043      text = Key.getString(m, "text");
044      command = Key.getString(m, "command");
045      option = Key.getString(m, "option");
046      select = Key.getString(m, "select");
047      toggle = Key.getString(m, "toggle");
048      label = Key.getString(m, "label");
049      preview = Key.getString(m, "preview");
050      parseSend(Key.getString(m, "send"));
051      parseLabel();
052      if (m.containsKey("states")) states = (List<String>)m.get("states");
053      sticky = (Boolean)Key.getValue(m, "sticky", false);
054      repeatable = (Boolean)Key.getValue(m, "repeatable", false);
055      functional = (Boolean)Key.getValue(m, "functional", true);
056    } else if (Key.androidKeys.contains(s)) {
057      code = Key.androidKeys.indexOf(s);
058      parseLabel();
059    } else {
060      text = s;
061      label = s;
062    }
063  }
064
065  private void parseSend(String s) {
066    if (s.isEmpty()) return;
067    String codes;
068    if (!s.contains("+")) codes = s;
069    else {
070      String[] ss = s.split("\\+");
071      int n = ss.length;
072      for (int i = 0; i < n - 1; i++) if (masks.containsKey(ss[i])) mask |= masks.get(ss[i]);
073      codes = ss[n - 1];
074    }
075    code = Key.androidKeys.indexOf(codes);
076  }
077
078  public String adjustCase(String s) {
079    if (s == null) return "";
080    if (s.length() == 1 && mKeyboard.isShifted()) s = s.toUpperCase();
081    return s;
082  }
083
084  public String getLabel() {
085    if (toggle != null && !toggle.isEmpty()) return states.get(Rime.getOption(toggle) ? 1 : 0);
086    return adjustCase(label);
087  }
088
089  public String getText() {
090    String s = "";
091    if (text != null && !text.isEmpty()) s = text;
092    else if (mKeyboard.isShifted() && mask == 0 && code >= KeyEvent.KEYCODE_A && code <= KeyEvent.KEYCODE_Z) s = label;
093    return adjustCase(s);
094  }
095
096  public String getPreviewText() {
097    if (preview != null && !preview.isEmpty()) return preview;
098    return getLabel();
099  }
100
101  public String getToggle() {
102    if (toggle != null && !toggle.isEmpty()) return toggle;
103    return "ascii_mode";
104  }
105
106  private void parseLabel() {
107    if (label != null && !label.isEmpty()) return;
108    int c = code;
109    if (c == KeyEvent.KEYCODE_SPACE){
110      label = Rime.getSchemaName();
111    } else {
112      if (c > 0) label = Key.androidKeys.get(c);
113    }
114  }
115
116  public int[] getRimeEvent() {
117    return getRimeEvent(code, mask);
118  }
119
120  public static int getRimeCode(int code) {
121    String s = Key.androidKeys.get(code);
122    return Rime.get_keycode_by_name(s);
123  }
124
125  public static int[] getRimeEvent(int code, int mask) {
126    int i = getRimeCode(code);
127    int m = 0;
128    if ((mask & KeyEvent.META_SHIFT_ON) > 0) {
129      m |= Rime.get_modifier_by_name("Shift");
130    }
131    if ((mask & KeyEvent.META_CTRL_ON) > 0) {
132      m |= Rime.get_modifier_by_name("Control");
133    }
134    if ((mask & KeyEvent.META_ALT_ON) > 0) {
135      m |= Rime.get_modifier_by_name("Alt");
136    }
137    if (mask == (1<<30)) m |= Rime.get_modifier_by_name("Release");
138    return new int[] {i, m};
139  }
140
141  public static Map<String,Integer> masks = new HashMap<String,Integer>() {
142    {
143      put("Shift", KeyEvent.META_SHIFT_ON);
144      put("Control", KeyEvent.META_CTRL_ON);
145      put("Alt", KeyEvent.META_ALT_ON);
146    }
147  };
148}
