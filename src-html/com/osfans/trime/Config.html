044  public Config(Context context) {
045    self = this;
046    maps = new HashMap<String, Map<String, Object>>();
047    mDefaultStyle = (Map<String,Object>)Rime.config_get_map("trime", "style");
048    fallbackColors = (Map<String,String>)Rime.config_get_map("trime", "fallback_colors");
049    Key.androidKeys = (List<String>)Rime.config_get_list("trime", "android_keys/name");
050    Key.presetKeys = (Map<String, Map>)Rime.config_get_map("trime", "preset_keys");
051    presetColorSchemes = Rime.config_get_map("trime", "preset_color_schemes");
052    presetKeyboards = Rime.config_get_map("trime", "preset_keyboards");
053    reset();
054  }

056  public static boolean prepareRime(Context context) {
057    if (new File(USER_DATA_DIR).exists()) return false;
058    copyFileOrDir(context, "rime", false);
059    Rime.get(true);
060    return true;
061  }

063  public static String[] list(Context context, String path) {
064    AssetManager assetManager = context.getAssets();
065    String assets[] = null;
066    try {
067      assets = assetManager.list(path);
068    } catch (IOException ex) {
069      Log.e("Config", "I/O Exception", ex);
070    }
071    return assets;
072  }
073
074  public static boolean copyFileOrDir(Context context, String path, boolean overwrite) {
075    AssetManager assetManager = context.getAssets();
076    String assets[] = null;
077    try {
078      assets = assetManager.list(path);
079      if (assets.length == 0) {
080        copyFile(context, path, overwrite);
081      } else {
082        String fullPath = "/sdcard/" + path;
083        File dir = new File(fullPath);
084        if (!dir.exists()) dir.mkdir();
085        for (int i = 0; i < assets.length; ++i) {
086          copyFileOrDir(context, path + "/" + assets[i], overwrite);
087        }
088      }
089    } catch (IOException ex) {
090      Log.e("Config", "I/O Exception", ex);
091      return false;
092    }
093    return true;
094  }
095
096  public static boolean copyFile(Context context, String filename, boolean overwrite) {
097    AssetManager assetManager = context.getAssets();
098    InputStream in = null;
099    OutputStream out = null;
100    try {
101      in = assetManager.open(filename);
102      String newFileName = "/sdcard/" + filename;
103      if (new File(newFileName).exists() && !overwrite) return true;
104      out = new FileOutputStream(newFileName);
105      byte[] buffer = new byte[BLK_SIZE];
106      int read;
107      while ((read = in.read(buffer)) != -1) {
108          out.write(buffer, 0, read);
109      }
110      in.close();
111      in = null;
112      out.flush();
113      out.close();
114      out = null;
115    } catch (Exception e) {
116      Log.e("Config", e.getMessage());
117      return false;
118    }
119    return true;
120  }

122  public void reset() {
123    schema_id = Rime.getSchemaId();
124    if (maps.containsKey(schema_id)) {
125      mStyle = maps.get(schema_id);
126      return;
127    }
128    mStyle = null;
129    File f = new File(USER_DATA_DIR, schema_id + "." + defaultName);
130    if (!f.exists()) return;
131    mStyle = (Map<String,Object>)Rime.config_get_map(schema_id + ".trime", "style");
132    maps.put(schema_id, mStyle); //緩存各方案自定義配置
133  }

135  private Object getValue(String k1) {
136    if (mStyle != null && mStyle.containsKey(k1)) return mStyle.get(k1);
137    if (mDefaultStyle != null && mDefaultStyle.containsKey(k1)) return mDefaultStyle.get(k1);
138    return null;
139  }

141  public String getKeyboardName(String name) {
142    if (name.contentEquals(".default")) {
143      if (presetKeyboards.containsKey(schema_id)) name = schema_id; //匹配方案名
144      else if (schema_id.indexOf("_") >= 0) {
145        name = schema_id.split("_")[0];
146        if (!presetKeyboards.containsKey(name)) { //匹配“_”前的方案名
147          Object o = Rime.schema_get_value(schema_id, "speller/alphabet");
148          name = "qwerty"; //26
149          if (o != null) {
150            String alphabet = (String) o;
151            if (presetKeyboards.containsKey(alphabet)) name = alphabet; //匹配字母表
152            else {
153              if (alphabet.indexOf(",") >= 0 || alphabet.indexOf(";") >= 0) name += "_";
154              if (alphabet.indexOf("0") >= 0 || alphabet.indexOf("1") >= 0) name += "0";
155            }
156          }
157        }
158      }
159    }
160    if (!presetKeyboards.containsKey(name)) name = "default";
161    Map<String, Object> m = (Map<String, Object>)presetKeyboards.get(name);
162    if (m.containsKey("import_preset")) {
163      name = (String)m.get("import_preset");
164    }
165    return name;
166  }

168  public List<String> getKeyboardNames() {
169    List<String> names = (List<String>)getValue("keyboards");
170    List<String> keyboards = new ArrayList<String>();
171    for (String s: names) {
172      s = getKeyboardName(s);
173      if (!keyboards.contains(s)) keyboards.add(s);
174    }
175    return keyboards;
176  }

178  public Map<String, Object> getKeyboard(String name) {
179    if (!presetKeyboards.containsKey(name)) name = "default";
180    return (Map<String, Object>)presetKeyboards.get(name);
181  }

195  public void destroy() {
196    if (maps != null) maps.clear();

198    if (mStyle != null) mStyle.clear();
199    self = null;

207  public double getDouble(String key) {
208    Object o = getValue(key);
209    double size = 0;
210    if (o instanceof Integer) size = ((Integer)o).doubleValue();
211    else if (o instanceof Float) size = ((Float)o).doubleValue();
212    else if (o instanceof Double) size = ((Double)o).doubleValue();
213    return size;
214  }

232  public int getColor(String key) {
233    String scheme = getString("color_scheme");
234    Map map = (Map<String, Object>)presetColorSchemes.get(scheme);
235    Object o = map.get(key);
236    String fallbackKey = key;
237    while (o == null && fallbackColors.containsKey(fallbackKey)) {
238      fallbackKey = fallbackColors.get(fallbackKey);
239      o = map.get(fallbackKey);
240    }
241    if (o == null) {
242      map = (Map<String, Object>)presetColorSchemes.get("default");
243      o = map.get(key);
244    }
245    if (o instanceof Integer) return ((Integer)o).intValue();
246    return ((Long)o).intValue();
247  }

255  public String[] getColorKeys() {
256    if (presetColorSchemes == null) return null;
257    String[] keys = new String[presetColorSchemes.size()];
258    presetColorSchemes.keySet().toArray(keys);
259    return keys;
260  }

262  public String[] getColorNames(String[] keys) {
263    if (keys == null) return null;
264    int n = keys.length;
265    String[] names = new String[n];
266    for (int i = 0; i < n; i++) {
267      Map<String, Object> m = (Map<String, Object>)presetColorSchemes.get(keys[i]);
268      names[i] = (String)m.get("name");
269    }
270    return names;
271  }

273  public Typeface getFont(String key){
274    String name = getString(key);
275    if (name != null) {
276      File f = new File(USER_DATA_DIR + "/fonts", name);
277      if(f.exists()) return Typeface.createFromFile(f);
278    }
279    return Typeface.DEFAULT;
280  }
