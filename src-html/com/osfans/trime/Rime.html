107  public static class RimeSchema {
108    private String kRightArrow = "→ ";
109    private String kRadioSelected = " ✓";
110
111    Map<String, Object> schema = new HashMap<String, Object>();
112    List<Map<String, Object>> switches = new ArrayList<Map<String, Object>>();
113
114    public RimeSchema(String schema_id) {
115      Object o;
116      o = schema_get_value(schema_id, "schema");
117      if (o == null || !(o instanceof Map)) return;
118      schema = (Map<String, Object>)o;
119      o = schema_get_value(schema_id, "switches");
120      if (o == null || !(o instanceof List)) return;
121      switches = (List<Map<String, Object>>)o;
122      check(); //檢查不在選單中顯示的選項
123    }
124
125    public void check() {
126      if (switches.isEmpty()) return;
127      int i = 0;
128      for (Map<String, Object> o: switches) {
129        if (!o.containsKey("states")) switches.remove(i);
130        i++;
131      }
132    }
133
134    public RimeCandidate[] getCandidates() {
135      if (switches.isEmpty()) return null;
136      RimeCandidate[] candidates = new RimeCandidate[switches.size()];
137      int i = 0;
138      for (Map<String, Object> o: switches) {
139        candidates[i] = new RimeCandidate();
140        List<String> states = (List<String>)o.get("states");
141        Integer value = (Integer)o.get("value");
142        if (value == null) value = 0;
143        candidates[i].text = states.get(value);
144        candidates[i].comment = o.containsKey("options") ? "" : kRightArrow + states.get(1 - value);
145        i++;
146      }
147      return candidates;
148    }
149
150    public void getValue(int session_id) {
151      if (switches.isEmpty()) return; //無方案
152      for (int j = 0; j < switches.size(); j++) {
153        Map<String, Object> o =  switches.get(j);
154        if (o.containsKey("options")) {
155          List<String> options = (List<String>)o.get("options");
156          for (int i = 0; i < options.size(); i++) {
157            String s = options.get(i);
158            if (Rime.get_option(session_id, s)) {
159              o.put("value", i);
160              break;
161            }
162          }
163        } else {
164          o.put("value", Rime.get_option(session_id, (String)o.get("name")) ? 1 : 0);
165        }
166        switches.set(j, o);
167      }
168    }
169
170    public void toggleOption(int i) {
171      if (switches.isEmpty()) return;
172      Map<String, Object> o =  switches.get(i);
173      Integer value = (Integer)o.get("value");
174      if (value == null) value = 0;
175      if (o.containsKey("options")) {
176        List<String> options = (List<String>)o.get("options");
177        Rime.setOption(options.get(value), false);
178        value = (value + 1) % options.size();
179        Rime.setOption(options.get(value), true);
180      } else {
181        value = 1 - value;
182        Rime.setOption((String)o.get("name"), value == 1);
183      }
184      o.put("value", value);
185      switches.set(i, o);
186    }
187  };

324  public static String[] getSelectLabels() {
325    if (mContext != null && mContext.size() > 0) {
326      if (mContext.select_labels != null) return mContext.select_labels;
327      if (mContext.menu.select_keys != null) return mContext.menu.select_keys.split("\\B");
328      int n = mContext.size();
329      String[] labels = new String[n];
330      for (int i = 0; i < n; i++) {
331        labels[i] = String.valueOf((i + 1)% 10);
332      }
333      return labels;
334    }
335    return null;
336  }

429  public static boolean selectSchema(int id) {
430    int n = mSchemaList.size();
431    if (id < 0 || id >= n) return false;
432    String schema_id = getSchemaId();
433    Map<String, String> m = (Map<String, String>)mSchemaList.get(id);
434    String target = m.get("schema_id");
435    if (target.contentEquals(schema_id)) return false;
436    return selectSchema(target);
437  }

462  public static void onMessage(int session_id, String message_type, String message_value) {
463    Log.info(String.format("message: [%d] [%s] %s", session_id, message_type, message_value));
464    Trime trime = Trime.getService();
465    switch (message_type) {
466      case "schema":
467        initSchema();
468        if (trime != null) {
469          trime.initKeyboard();
470          trime.updateComposing();
471        }
472        break;
473      case "option":
474        getStatus();
475        if (trime != null) {
476          trime.invalidateKeyboard(); //鍵盤狀態
477          if (message_value.endsWith("ascii_mode")) trime.setLanguage(isAsciiMode());
478        }
479        break;
480    }
481  }
