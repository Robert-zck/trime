017package com.osfans.trime;
018
019import android.content.Context;
020import android.content.Intent;
021
022import java.text.SimpleDateFormat;
023import java.util.Date;
024
025public class Function {
026  private static String TAG = "Function";
027
028  public static void openApp(Context context, String s) {
029    Intent intent = context.getPackageManager().getLaunchIntentForPackage(s);
030    intent.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK | Intent.FLAG_ACTIVITY_NO_HISTORY);
031    context.startActivity(intent);
032  }
033
034  public static void showPrefDialog(Context context) {
035    Intent intent = new Intent(context, Pref.class);
036    intent.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK | Intent.FLAG_ACTIVITY_NO_HISTORY);
037    context.startActivity(intent);
038  }
039
040  public static String handle(Context context, String command, String option) {
041    String s = null;
042    if (command == null) return s;
043    switch (command) {
044      case "date":
045        s = new SimpleDateFormat(option).format(new Date()); //時間
046        break;
047      case "run":
048        openApp(context, option); //啓動程序
049      default:
050        break;
051    }
052    return s;
053  }
054}
