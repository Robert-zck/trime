117    public Keyboard(Context context) {
118      DisplayMetrics dm = context.getResources().getDisplayMetrics();
119      mDisplayWidth = dm.widthPixels;
120      mDisplayHeight = dm.heightPixels;
121      //Log.v(TAG, "keyboard's display metrics:" + dm);
122
123      Config config = Config.get();
124      mDefaultHorizontalGap = config.getPixel("horizontal_gap");
125      mDefaultVerticalGap = config.getPixel("vertical_gap");
126      mDefaultWidth = (int)(mDisplayWidth * config.getDouble("key_width") / 100);
127      mDefaultHeight = config.getPixel("key_height");
128      mProximityThreshold = (int) (mDefaultWidth * SEARCH_DISTANCE);
129      mProximityThreshold = mProximityThreshold * mProximityThreshold; // Square it for comparison
130
131      mKeys = new ArrayList<Key>();
132      mComposingKeys = new ArrayList<Key>();
133    }

135    /**
136     * <p>Creates a blank keyboard from the given resource file and populates it with the specified
137     * characters in left-to-right, top-to-bottom fashion, using the specified number of columns.
138     * </p>
139     * <p>If the specified number of columns is -1, then the keyboard will fit as many keys as
140     * possible in each row.</p>
141     * @param context the application or service context
142     * @param layoutTemplateResId the layout template file, containing no keys.
143     * @param characters the list of characters to display on the keyboard. One key will be created
144     * for each character.
145     * @param columns the number of columns of keys to display. If this number is greater than the 
146     * number of keys that can fit in a row, it will be ignored. If this number is -1, the 
147     * keyboard will fit as many keys as possible in each row.
148     */
149    public Keyboard(Context context, CharSequence characters, int columns, int horizontalPadding) {
150        this(context);
151        int x = 0;
152        int y = 0;
153        int column = 0;
154        mTotalWidth = 0;
155
156        final int maxColumns = columns == -1 ? Integer.MAX_VALUE : columns;
157        for (int i = 0; i < characters.length(); i++) {
158            char c = characters.charAt(i);
159            if (column >= maxColumns 
160                    || x + mDefaultWidth + horizontalPadding > mDisplayWidth) {
161                x = 0;
162                y += mDefaultVerticalGap + mDefaultHeight;
163                column = 0;
164            }
165            final Key key = new Key(this);
166            key.x = x;
167            key.y = y;
168            key.width = mDefaultWidth;
169            key.height = mDefaultHeight;
170            key.gap = mDefaultHorizontalGap;
171            key.events[0] = new Event(this, String.valueOf(c));
172            column++;
173            x += key.width + key.gap;
174            mKeys.add(key);
175            if (x > mTotalWidth) {
176                mTotalWidth = x;
177            }
178        }
179        mTotalHeight = y + mDefaultHeight; 
180    }

242  public boolean toggleModifier(int mask) {
243    boolean value = !hasModifier(mask);
244    if (value) mMetaState |= mask;
245    else mMetaState &= ~mask;
246    return value;
247  }

253  public boolean setModifier(int mask, boolean value) {
254    boolean b = hasModifier(mask);
255    if (b == value) return false;
256    if (value) mMetaState |= mask;
257    else mMetaState &= ~mask;
258    return true;
259  }

283    private void computeNearestNeighbors() {
284        // Round-up so we don't have any pixels outside the grid
285        mCellWidth = (getMinWidth() + GRID_WIDTH - 1) / GRID_WIDTH;
286        mCellHeight = (getHeight() + GRID_HEIGHT - 1) / GRID_HEIGHT;
287        mGridNeighbors = new int[GRID_SIZE][];
288        int[] indices = new int[mKeys.size()];
289        final int gridWidth = GRID_WIDTH * mCellWidth;
290        final int gridHeight = GRID_HEIGHT * mCellHeight;
291        for (int x = 0; x < gridWidth; x += mCellWidth) {
292            for (int y = 0; y < gridHeight; y += mCellHeight) {
293                int count = 0;
294                for (int i = 0; i < mKeys.size(); i++) {
295                    final Key key = mKeys.get(i);
296                    if (key.squaredDistanceFrom(x, y) < mProximityThreshold ||
297                            key.squaredDistanceFrom(x + mCellWidth - 1, y) < mProximityThreshold ||
298                            key.squaredDistanceFrom(x + mCellWidth - 1, y + mCellHeight - 1) 
299                                < mProximityThreshold ||
300                            key.squaredDistanceFrom(x, y + mCellHeight - 1) < mProximityThreshold) {
301                        indices[count++] = i;
302                    }
303                }
304                int [] cell = new int[count];
305                System.arraycopy(indices, 0, cell, 0, count);
306                mGridNeighbors[(y / mCellHeight) * GRID_WIDTH + (x / mCellWidth)] = cell;
307            }
308        }
309    }

311    /**
312     * Returns the indices of the keys that are closest to the given point.
313     * @param x the x-coordinate of the point
314     * @param y the y-coordinate of the point
315     * @return the array of integer indices for the nearest keys to the given point. If the given
316     * point is out of range, then an array of size zero is returned.
317     */
318    public int[] getNearestKeys(int x, int y) {
319        if (mGridNeighbors == null) computeNearestNeighbors();
320        if (x >= 0 && x < getMinWidth() && y >= 0 && y < getHeight()) {
321            int index = (y / mCellHeight) * GRID_WIDTH + (x / mCellWidth);
322            if (index < GRID_SIZE) {
323                return mGridNeighbors[index];
324            }
325        }
326        return new int[0];
327    }

329  public Keyboard(Context context, String name) {
330    this(context);
331    Map<String,Object> m = Config.get().getKeyboard(name);
332    mAsciiMode = (Integer)Key.getValue(m, "ascii_mode", 1);
333    int columns = (Integer)Key.getValue(m, "columns", 20);
334    int defaultWidth = (int)(Key.getDouble(m, "width", 0) * mDisplayWidth / 100);
335    if (defaultWidth == 0) defaultWidth = mDefaultWidth;
336    double height = Key.getDouble(m, "height", 0);
337    int defaultHeight = mDefaultHeight;
338    if (height > 0) defaultHeight = (int)TypedValue.applyDimension(TypedValue.COMPLEX_UNIT_SP, (float)height, Resources.getSystem().getDisplayMetrics());
339    List<Map<String,Object>> lm = (List<Map<String,Object>>)m.get("keys");
340
341    int x = 0;
342    int y = 0;
343    int column = 0;
344    mTotalWidth = 0;
345
346    final int maxColumns = columns == -1 ? Integer.MAX_VALUE : columns;
347    for (Map<String,Object> mk: lm) {
348      int gap = mDefaultHorizontalGap;
349      int w = (int)(Key.getDouble(mk, "width", 0) * mDisplayWidth / 100);
350      if (w == 0 && mk.containsKey("click")) w = defaultWidth;
351      w -= gap;
352      if (column >= maxColumns || x + w > mDisplayWidth) {
353        x = 0;
354        y += mDefaultVerticalGap + defaultHeight;
355        column = 0;
356      }
357      if (!mk.containsKey("click")){ //無按鍵事件
358        x += w + gap;
359        continue; //縮進
360      }
361
362      final Key key = new Key(this, mk);
363      key.x = x;
364      key.y = y;
365      int right_gap = Math.abs(mDisplayWidth - x - w - gap);
366      key.width = (right_gap <= mDisplayWidth / 100) ? mDisplayWidth - x : w; //右側不留白
367      key.height = defaultHeight;
368      key.gap = gap;
369      column++;
370      x += key.width + key.gap;
371      mKeys.add(key);
372      if (x > mTotalWidth) {
373          mTotalWidth = x;
374      }
375    }
376    mTotalHeight = y + defaultHeight;
377  }
