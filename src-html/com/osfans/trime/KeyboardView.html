082    public interface OnKeyboardActionListener {
083        
084        /**
085         * Called when the user presses a key. This is sent before the {@link #onKey} is called.
086         * For keys that repeat, this is only called once.
087         * @param primaryCode the unicode of the key being pressed. If the touch is not on a valid
088         * key, the value will be zero.
089         */
090        void onPress(int primaryCode);
091        
092        /**
093         * Called when the user releases a key. This is sent after the {@link #onKey} is called.
094         * For keys that repeat, this is only called once.
095         * @param primaryCode the code of the key that was released
096         */
097        void onRelease(int primaryCode);
098
099        /**
100         * Send a key press to the listener.
101         * @param primaryCode this is the key that was pressed
102         * @param keyCodes the codes for all the possible alternative keys
103         * with the primary code being the first. If the primary key code is
104         * a single character such as an alphabet or number or symbol, the alternatives
105         * will include other characters that may be on the same key or adjacent keys.
106         * These codes are useful to correct for accidental presses of a key adjacent to
107         * the intended key.
108         */
109        void onEvent(Event event);
110        void onKey(int primaryCode, int mask);
111
112        /**
113         * Sends a sequence of characters to the listener.
114         * @param text the sequence of characters to be displayed.
115         */
116        void onText(CharSequence text);
117        
118        /**
119         * Called when the user quickly moves the finger from right to left.
120         */
121        void swipeLeft();
122        
123        /**
124         * Called when the user quickly moves the finger from left to right.
125         */
126        void swipeRight();
127        
128        /**
129         * Called when the user quickly moves the finger from up to down.
130         */
131        void swipeDown();
132        
133        /**
134         * Called when the user quickly moves the finger from down to up.
135         */
136        void swipeUp();
137    }

264    Handler mHandler = new Handler() {
265        @Override
266        public void handleMessage(Message msg) {
267            switch (msg.what) {
268                case MSG_SHOW_PREVIEW:
269                    showKey(msg.arg1, msg.arg2);
270                    break;
271                case MSG_REMOVE_PREVIEW:
272                    mPreviewText.setVisibility(INVISIBLE);
273                    break;
274                case MSG_REPEAT:
275                    if (repeatKey()) {
276                        Message repeat = Message.obtain(this, MSG_REPEAT);
277                        sendMessageDelayed(repeat, REPEAT_INTERVAL);                        
278                    }
279                    break;
280                case MSG_LONGPRESS:
281                    openPopupIfRequired((MotionEvent) msg.obj);
282                    break;
283            }
284        }
285    };

384    private void initGestureDetector() {
385        mGestureDetector = new GestureDetector(getContext(), new GestureDetector.SimpleOnGestureListener() {
386            @Override
387            public boolean onFling(MotionEvent me1, MotionEvent me2, 
388                    float velocityX, float velocityY) {
389                if (mPossiblePoly) return false;
390                final float absX = Math.abs(velocityX);
391                final float absY = Math.abs(velocityY);
392                float deltaX = me2.getX() - me1.getX();
393                float deltaY = me2.getY() - me1.getY();
394                int travelX = 0; //getWidth() / 2; // Half the keyboard width
395                int travelY = 0; //getHeight() / 2; // Half the keyboard height
396                mSwipeTracker.computeCurrentVelocity(10);
397                final float endingVelocityX = mSwipeTracker.getXVelocity();
398                final float endingVelocityY = mSwipeTracker.getYVelocity();
399                boolean sendDownKey = false;
400                int type = 0;
401                if (velocityX > mSwipeThreshold && absY < absX && deltaX > travelX) {
402                    if (mDisambiguateSwipe && endingVelocityX < velocityX / 4) {
403                        sendDownKey = true;
404                        type = Key.SWIPE_RIGHT;
405                    } else {
406                        swipeRight();
407                        return true;
408                    }
409                } else if (velocityX < -mSwipeThreshold && absY < absX && deltaX < -travelX) {
410                    if (mDisambiguateSwipe && endingVelocityX > velocityX / 4) {
411                        sendDownKey = true;
412                        type = Key.SWIPE_LEFT;
413                    } else {
414                        swipeLeft();
415                        return true;
416                    }
417                } else if (velocityY < -mSwipeThreshold && absX < absY && deltaY < -travelY) {
418                    if (mDisambiguateSwipe && endingVelocityY > velocityY / 4) {
419                        sendDownKey = true;
420                        type = Key.SWIPE_UP;
421                    } else {
422                        swipeUp();
423                        return true;
424                    }
425                } else if (velocityY > mSwipeThreshold && absX < absY / 2 && deltaY > travelY) {
426                    if (mDisambiguateSwipe && endingVelocityY < velocityY / 4) {
427                        sendDownKey = true;
428                        type = Key.SWIPE_DOWN;
429                    } else {
430                        swipeDown();
431                        return true;
432                    }
433                }
434
435                if (sendDownKey) {
436                    showPreview(NOT_A_KEY);
437                    showPreview(mDownKey, type);
438                    detectAndSendKey(mDownKey, mStartX, mStartY, me1.getEventTime(), type);
439                    return true;
440                }
441                return false;
442            }
443        });
444
445        mGestureDetector.setIsLongpressEnabled(false);
446    }

460    /**
461     * Attaches a keyboard to this view. The keyboard can be switched at any time and the
462     * view will re-layout itself to accommodate the keyboard.
463     * @see Keyboard
464     * @see #getKeyboard()
465     * @param keyboard the keyboard to display in this view
466     */
467    public void setKeyboard(Keyboard keyboard) {
468        if (mKeyboard != null) {
469            showPreview(NOT_A_KEY);
470        }
471        // Remove any pending messages
472        removeMessages();
473        mKeyboard = keyboard;
474        List<Key> keys = mKeyboard.getKeys();
475        mKeys = keys.toArray(new Key[keys.size()]);
476        requestLayout();
477        // Hint to reallocate the buffer if the size changed
478        mKeyboardChanged = true;
479        invalidateAllKeys();
480        computeProximityThreshold(keyboard);
481        mMiniKeyboardCache.clear(); // Not really necessary to do every time, but will free up views
482        // Switching to a different keyboard should abort any pending keys so that the key up
483        // doesn't get delivered to the old or new keyboard
484        mAbortKey = true; // Until the next ACTION_DOWN
485    }

610    /**
611     * Compute the average distance between adjacent keys (horizontally and vertically)
612     * and square it to get the proximity threshold. We use a square here and in computing
613     * the touch distance from a key's center to avoid taking a square root.
614     * @param keyboard
615     */
616    private void computeProximityThreshold(Keyboard keyboard) {
617        if (keyboard == null) return;
618        final Key[] keys = mKeys;
619        if (keys == null) return;
620        int length = keys.length;
621        int dimensionSum = 0;
622        for (int i = 0; i < length; i++) {
623            Key key = keys[i];
624            dimensionSum += Math.min(key.width, key.height) + key.gap;
625        }
626        if (dimensionSum < 0 || length == 0) return;
627        mProximityThreshold = (int) (dimensionSum * 1.4f / length);
628        mProximityThreshold *= mProximityThreshold; // Square it
629    }

650    private void onBufferDraw() {
651        if (mBuffer == null || mKeyboardChanged) {
652            if (mBuffer == null || mKeyboardChanged &&
653                    (mBuffer.getWidth() != getWidth() || mBuffer.getHeight() != getHeight())) {
654                // Make sure our bitmap is at least 1x1
655                final int width = Math.max(1, getWidth());
656                final int height = Math.max(1, getHeight());
657                mBuffer = Bitmap.createBitmap(width, height, Bitmap.Config.ARGB_8888);
658                mCanvas = new Canvas(mBuffer);
659            }
660            invalidateAllKeys();
661            mKeyboardChanged = false;
662        }
663        final Canvas canvas = mCanvas;
664        canvas.clipRect(mDirtyRect, Op.REPLACE);
665        
666        if (mKeyboard == null) return;
667        
668        final Paint paint = mPaint;
669        Drawable keyBackground;
670        final Rect clipRegion = mClipRegion;
671        final Rect padding = mPadding;
672        final int kbdPaddingLeft = getPaddingLeft();
673        final int kbdPaddingTop = getPaddingTop();
674        final Key[] keys = mKeys;
675        final Key invalidKey = mInvalidatedKey;
676
677        boolean drawSingleKey = false;
678        if (invalidKey != null && canvas.getClipBounds(clipRegion)) {
679          // Is clipRegion completely contained within the invalidated key?
680          if (invalidKey.x + kbdPaddingLeft - 1 <= clipRegion.left &&
681                  invalidKey.y + kbdPaddingTop - 1 <= clipRegion.top &&
682                  invalidKey.x + invalidKey.width + kbdPaddingLeft + 1 >= clipRegion.right &&
683                  invalidKey.y + invalidKey.height + kbdPaddingTop + 1 >= clipRegion.bottom) {
684              drawSingleKey = true;
685          }
686        }
687        canvas.drawColor(0x00000000, PorterDuff.Mode.CLEAR);
688        final int keyCount = keys.length;
689        final float symbolTop = (mPaintSymbol.getTextSize());//mPaintSymbol.descent();
690        for (int i = 0; i < keyCount; i++) {
691            final Key key = keys[i];
692            if (drawSingleKey && invalidKey != key) {
693                continue;
694            }
695            int[] drawableState = key.getCurrentDrawableState();
696            keyBackground = new PaintDrawable(mKeyBackColor.getColorForState(drawableState, 0));
697            ((PaintDrawable)keyBackground).setCornerRadius(mRoundCorner);
698            mPaint.setColor(mKeyTextColor.getColorForState(drawableState, 0));
699            mPaintSymbol.setColor(key.pressed ? hilited_key_symbol_color: key_symbol_color);
700
701            // Switch the character to uppercase if shift is pressed
702            String label = key.getLabel();
703            String hint = key.hint;
704            int left = (key.width - padding.left - padding.right) / 2 + padding.left;
705            int top = padding.top;
706            
707            final Rect bounds = keyBackground.getBounds();
708            if (key.width != bounds.right || 
709                    key.height != bounds.bottom) {
710                keyBackground.setBounds(0, 0, key.width, key.height);
711            }
712            canvas.translate(key.x + kbdPaddingLeft, key.y + kbdPaddingTop);
713            keyBackground.draw(canvas);
714            
715            if (!label.isEmpty()) {
716                // For characters, use large font. For labels like "Done", use small font.
717                if (label.length() > 1) {
718                    paint.setTextSize(mLabelTextSize);
719                } else {
720                    paint.setTextSize(mKeyTextSize);
721                }
722                // Draw a drop shadow for the text
723                paint.setShadowLayer(mShadowRadius, 0, 0, mShadowColor);
724                // Draw the text
725                canvas.drawText(label, left,
726                    (key.height - padding.top - padding.bottom) / 2
727                            + (paint.getTextSize() - paint.descent()) / 2 + top,
728                    paint);
729
730                if (key.getLongClick() != null) {
731                    mPaintSymbol.setShadowLayer(mShadowRadius, 0, 0, mShadowColor);
732                    canvas.drawText(key.getSymbolLabel(), left, symbolTop + top, mPaintSymbol);
733                }
734
735                if (!hint.isEmpty()) {
736                    mPaintSymbol.setShadowLayer(mShadowRadius, 0, 0, mShadowColor);
737                    canvas.drawText(hint, left, key.height  - padding.bottom - symbolTop + top, mPaintSymbol);
738                }
739
740                // Turn off drop shadow
741                paint.setShadowLayer(0, 0, 0, 0);
742            }
743            canvas.translate(-key.x - kbdPaddingLeft, -key.y - kbdPaddingTop);
744        }
745        mInvalidatedKey = null;
746        // Overlay a dark rectangle to dim the keyboard
747        if (mMiniKeyboardOnScreen) {
748            paint.setColor((int) (mBackgroundDimAmount * 0xFF) << 24);
749            canvas.drawRect(0, 0, getWidth(), getHeight(), paint);
750        }
751
752        if (DEBUG && mShowTouchPoints) {
753            paint.setAlpha(128);
754            paint.setColor(0xFFFF0000);
755            canvas.drawCircle(mStartX, mStartY, 3, paint);
756            canvas.drawLine(mStartX, mStartY, mLastX, mLastY, paint);
757            paint.setColor(0xFF0000FF);
758            canvas.drawCircle(mLastX, mLastY, 3, paint);
759            paint.setColor(0xFF00FF00);
760            canvas.drawCircle((mStartX + mLastX) / 2, (mStartY + mLastY) / 2, 2, paint);
761        }
762        
763        mDrawPending = false;
764        mDirtyRect.setEmpty();
765    }

767    private int getKeyIndices(int x, int y, int[] allKeys) {
768        final Key[] keys = mKeys;
769        int primaryIndex = NOT_A_KEY;
770        int closestKey = NOT_A_KEY;
771        int closestKeyDist = mProximityThreshold + 1;
772        java.util.Arrays.fill(mDistances, Integer.MAX_VALUE);
773        int [] nearestKeyIndices = mKeyboard.getNearestKeys(x, y);
774        final int keyCount = nearestKeyIndices.length;
775        for (int i = 0; i < keyCount; i++) {
776            final Key key = keys[nearestKeyIndices[i]];
777            int dist = 0;
778            boolean isInside = key.isInside(x,y);
779            if (isInside) {
780                primaryIndex = nearestKeyIndices[i];
781            }
782
783            if (((mProximityCorrectOn 
784                    && (dist = key.squaredDistanceFrom(x, y)) < mProximityThreshold) 
785                    || isInside)
786                    && key.getCode() > 32) {
787                // Find insertion point
788                final int nCodes = 1;
789                if (dist < closestKeyDist) {
790                    closestKeyDist = dist;
791                    closestKey = nearestKeyIndices[i];
792                }
793                
794                if (allKeys == null) continue;
795                
796                for (int j = 0; j < mDistances.length; j++) {
797                    if (mDistances[j] > dist) {
798                        // Make space for nCodes codes
799                        System.arraycopy(mDistances, j, mDistances, j + nCodes,
800                                mDistances.length - j - nCodes);
801                        System.arraycopy(allKeys, j, allKeys, j + nCodes,
802                                allKeys.length - j - nCodes);
803                        allKeys[j] = key.getCode();
804                        mDistances[j] = dist;
805                        break;
806                    }
807                }
808            }
809        }
810        if (primaryIndex == NOT_A_KEY) {
811            primaryIndex = closestKey;
812        }
813        return primaryIndex;
814    }

840    private void showPreview(int keyIndex, int type) {
841        int oldKeyIndex = mCurrentKeyIndex;
842        final PopupWindow previewPopup = mPreviewPopup;
843        
844        mCurrentKeyIndex = keyIndex;
845        // Release the old key and press the new key
846        final Key[] keys = mKeys;
847        if (oldKeyIndex != mCurrentKeyIndex) {
848            if (oldKeyIndex != NOT_A_KEY && keys.length > oldKeyIndex) {
849                Key oldKey = keys[oldKeyIndex];
850                oldKey.onReleased(mCurrentKeyIndex == NOT_A_KEY);
851                invalidateKey(oldKeyIndex);
852            }
853            if (mCurrentKeyIndex != NOT_A_KEY && keys.length > mCurrentKeyIndex) {
854                Key newKey = keys[mCurrentKeyIndex];
855                newKey.onPressed();
856                invalidateKey(mCurrentKeyIndex);
857            }
858        }
859        // If key changed and preview is on ...
860        if (oldKeyIndex != mCurrentKeyIndex && mShowPreview) {
861            mHandler.removeMessages(MSG_SHOW_PREVIEW);
862            if (previewPopup.isShowing()) {
863                if (keyIndex == NOT_A_KEY) {
864                    mHandler.sendMessageDelayed(mHandler
865                            .obtainMessage(MSG_REMOVE_PREVIEW),
866                            DELAY_AFTER_PREVIEW);
867                }
868            }
869            if (keyIndex != NOT_A_KEY) {
870                if (previewPopup.isShowing() && mPreviewText.getVisibility() == VISIBLE) {
871                    // Show right away, if it's already visible and finger is moving around
872                    showKey(keyIndex, type);
873                } else {
874                    mHandler.sendMessageDelayed(
875                            mHandler.obtainMessage(MSG_SHOW_PREVIEW, keyIndex, type),
876                            DELAY_BEFORE_PREVIEW);
877                }
878            }
879        }
880    }

887    private void showKey(final int keyIndex, int type) {
888        final PopupWindow previewPopup = mPreviewPopup;
889        final Key[] keys = mKeys;
890        if (keyIndex < 0 || keyIndex >= mKeys.length) return;
891        Key key = keys[keyIndex];
892        mPreviewText.setCompoundDrawables(null, null, null, null);
893        mPreviewText.setText(key.getPreviewText(type));
894        mPreviewText.measure(MeasureSpec.makeMeasureSpec(0, MeasureSpec.UNSPECIFIED), 
895                MeasureSpec.makeMeasureSpec(0, MeasureSpec.UNSPECIFIED));
896        int popupWidth = Math.max(mPreviewText.getMeasuredWidth(), key.width 
897                + mPreviewText.getPaddingLeft() + mPreviewText.getPaddingRight());
898        final int popupHeight = mPreviewHeight;
899        LayoutParams lp = mPreviewText.getLayoutParams();
900        if (lp != null) {
901            lp.width = popupWidth;
902            lp.height = popupHeight;
903        }
904        if (!mPreviewCentered) {
905            mPopupPreviewX = key.x - mPreviewText.getPaddingLeft() + getPaddingLeft();
906            mPopupPreviewY = key.y - popupHeight + mPreviewOffset;
907        } else {
908            // TODO: Fix this if centering is brought back
909            mPopupPreviewX = 160 - mPreviewText.getMeasuredWidth() / 2;
910            mPopupPreviewY = - mPreviewText.getMeasuredHeight();
911        }
912        mHandler.removeMessages(MSG_REMOVE_PREVIEW);
913        getLocationInWindow(mCoordinates);
914        mCoordinates[0] += mMiniKeyboardOffsetX; // Offset may be zero
915        mCoordinates[1] += mMiniKeyboardOffsetY; // Offset may be zero
916
917        // Set the preview background state
918        mPreviewText.getBackground().setState(
919                key.popupResId != 0 ? LONG_PRESSABLE_STATE_SET : EMPTY_STATE_SET);
920        mPopupPreviewX += mCoordinates[0];
921        mPopupPreviewY += mCoordinates[1];
922
923        // If the popup cannot be shown above the key, put it on the side
924        getLocationOnScreen(mCoordinates);
925        if (mPopupPreviewY + mCoordinates[1] < 0) {
926            // If the key you're pressing is on the left side of the keyboard, show the popup on
927            // the right, offset by enough to see at least one key to the left/right.
928            if (key.x + key.width <= getWidth() / 2) {
929                mPopupPreviewX += (int) (key.width * 2.5);
930            } else {
931                mPopupPreviewX -= (int) (key.width * 2.5);
932            }
933            mPopupPreviewY += popupHeight;
934        }
935
936        if (previewPopup.isShowing()) {
937            previewPopup.update(mPopupPreviewX, mPopupPreviewY,
938                    popupWidth, popupHeight);
939        } else {
940            previewPopup.setWidth(popupWidth);
941            previewPopup.setHeight(popupHeight);
942            previewPopup.showAtLocation(mPopupParent, Gravity.NO_GRAVITY, 
943                    mPopupPreviewX, mPopupPreviewY);
944        }
945        mPreviewText.setVisibility(VISIBLE);
946    }

1135    @Override
1136    public boolean onTouchEvent(MotionEvent me) {
1137        // Convert multi-pointer up/down events to single up/down events to 
1138        // deal with the typical multi-pointer behavior of two-thumb typing
1139        final int pointerCount = me.getPointerCount();
1140        final int action = me.getAction();
1141        boolean result = false;
1142        final long now = me.getEventTime();
1143
1144        if (pointerCount != mOldPointerCount) {
1145            if (pointerCount == 1) {
1146                // Send a down event for the latest pointer
1147                MotionEvent down = MotionEvent.obtain(now, now, MotionEvent.ACTION_DOWN,
1148                        me.getX(), me.getY(), me.getMetaState());
1149                result = onModifiedTouchEvent(down, false);
1150                down.recycle();
1151                // If it's an up action, then deliver the up as well.
1152                if (action == MotionEvent.ACTION_UP) {
1153                    result = onModifiedTouchEvent(me, true);
1154                }
1155            } else {
1156                // Send an up event for the last pointer
1157                MotionEvent up = MotionEvent.obtain(now, now, MotionEvent.ACTION_UP,
1158                        mOldPointerX, mOldPointerY, me.getMetaState());
1159                result = onModifiedTouchEvent(up, true);
1160                up.recycle();
1161            }
1162        } else {
1163            if (pointerCount == 1) {
1164                result = onModifiedTouchEvent(me, false);
1165                mOldPointerX = me.getX();
1166                mOldPointerY = me.getY();
1167            } else {
1168                // Don't do anything when 2 pointers are down and moving.
1169                result = true;
1170            }
1171        }
1172        mOldPointerCount = pointerCount;
1173
1174        return result;
1175    }

1177    private boolean onModifiedTouchEvent(MotionEvent me, boolean possiblePoly) {
1178        int touchX = (int) me.getX() - getPaddingLeft();
1179        int touchY = (int) me.getY() - getPaddingTop();
1180        if (touchY >= -mVerticalCorrection)
1181            touchY += mVerticalCorrection;
1182        final int action = me.getAction();
1183        final long eventTime = me.getEventTime();
1184        int keyIndex = getKeyIndices(touchX, touchY, null);
1185        mPossiblePoly = possiblePoly;
1186
1187        // Track the last few movements to look for spurious swipes.
1188        if (action == MotionEvent.ACTION_DOWN) mSwipeTracker.clear();
1189        mSwipeTracker.addMovement(me);
1190
1191        // Ignore all motion events until a DOWN.
1192        if (mAbortKey
1193                && action != MotionEvent.ACTION_DOWN && action != MotionEvent.ACTION_CANCEL) {
1194            return true;
1195        }
1196
1197        if (mGestureDetector.onTouchEvent(me)) {
1198            showPreview(NOT_A_KEY);
1199            mHandler.removeMessages(MSG_REPEAT);
1200            mHandler.removeMessages(MSG_LONGPRESS);
1201            return true;
1202        }
1203        
1204        // Needs to be called after the gesture detector gets a turn, as it may have
1205        // displayed the mini keyboard
1206        if (mMiniKeyboardOnScreen && action != MotionEvent.ACTION_CANCEL) {
1207            return true;
1208        }
1209
1210        switch (action) {
1211            case MotionEvent.ACTION_DOWN:
1212                mAbortKey = false;
1213                mStartX = touchX;
1214                mStartY = touchY;
1215                mLastCodeX = touchX;
1216                mLastCodeY = touchY;
1217                mLastKeyTime = 0;
1218                mCurrentKeyTime = 0;
1219                mLastKey = NOT_A_KEY;
1220                mCurrentKey = keyIndex;
1221                mDownKey = keyIndex;
1222                mDownTime = me.getEventTime();
1223                mLastMoveTime = mDownTime;
1224                checkMultiTap(eventTime, keyIndex);
1225                mKeyboardActionListener.onPress(keyIndex != NOT_A_KEY ? 
1226                        mKeys[keyIndex].getCode() : 0);
1227                if (mCurrentKey >= 0 && mKeys[mCurrentKey].getClick().repeatable) {
1228                    mRepeatKeyIndex = mCurrentKey;
1229                    Message msg = mHandler.obtainMessage(MSG_REPEAT);
1230                    mHandler.sendMessageDelayed(msg, REPEAT_START_DELAY);
1231                    repeatKey();
1232                    // Delivering the key could have caused an abort
1233                    if (mAbortKey) {
1234                        mRepeatKeyIndex = NOT_A_KEY;
1235                        break;
1236                    }
1237                }
1238                if (mCurrentKey != NOT_A_KEY) {
1239                    Message msg = mHandler.obtainMessage(MSG_LONGPRESS, me);
1240                    mHandler.sendMessageDelayed(msg, LONGPRESS_TIMEOUT);
1241                }
1242                showPreview(keyIndex, 0);
1243                break;
1244
1245            case MotionEvent.ACTION_MOVE:
1246                boolean continueLongPress = false;
1247                if (keyIndex != NOT_A_KEY) {
1248                    if (mCurrentKey == NOT_A_KEY) {
1249                        mCurrentKey = keyIndex;
1250                        mCurrentKeyTime = eventTime - mDownTime;
1251                    } else {
1252                        if (keyIndex == mCurrentKey) {
1253                            mCurrentKeyTime += eventTime - mLastMoveTime;
1254                            continueLongPress = true;
1255                        } else if (mRepeatKeyIndex == NOT_A_KEY) {
1256                            resetMultiTap();
1257                            mLastKey = mCurrentKey;
1258                            mLastCodeX = mLastX;
1259                            mLastCodeY = mLastY;
1260                            mLastKeyTime =
1261                                    mCurrentKeyTime + eventTime - mLastMoveTime;
1262                            mCurrentKey = keyIndex;
1263                            mCurrentKeyTime = 0;
1264                        }
1265                    }
1266                }
1267                if (!continueLongPress) {
1268                    // Cancel old longpress
1269                    mHandler.removeMessages(MSG_LONGPRESS);
1270                    // Start new longpress if key has changed
1271                    if (keyIndex != NOT_A_KEY) {
1272                        Message msg = mHandler.obtainMessage(MSG_LONGPRESS, me);
1273                        mHandler.sendMessageDelayed(msg, LONGPRESS_TIMEOUT);
1274                    }
1275                }
1276                showPreview(mCurrentKey);
1277                mLastMoveTime = eventTime;
1278                break;
1279
1280            case MotionEvent.ACTION_UP:
1281                removeMessages();
1282                if (keyIndex == mCurrentKey) {
1283                    mCurrentKeyTime += eventTime - mLastMoveTime;
1284                } else {
1285                    resetMultiTap();
1286                    mLastKey = mCurrentKey;
1287                    mLastKeyTime = mCurrentKeyTime + eventTime - mLastMoveTime;
1288                    mCurrentKey = keyIndex;
1289                    mCurrentKeyTime = 0;
1290                }
1291                if (mCurrentKeyTime < mLastKeyTime && mCurrentKeyTime < DEBOUNCE_TIME
1292                        && mLastKey != NOT_A_KEY) {
1293                    mCurrentKey = mLastKey;
1294                    touchX = mLastCodeX;
1295                    touchY = mLastCodeY;
1296                }
1297                showPreview(NOT_A_KEY);
1298                Arrays.fill(mKeyIndices, NOT_A_KEY);
1299                // If we're not on a repeating key (which sends on a DOWN event)
1300                if (mRepeatKeyIndex == NOT_A_KEY && !mMiniKeyboardOnScreen && !mAbortKey) {
1301                    detectAndSendKey(mCurrentKey, touchX, touchY, eventTime);
1302                }
1303                invalidateKey(keyIndex);
1304                mRepeatKeyIndex = NOT_A_KEY;
1305                break;
1306            case MotionEvent.ACTION_CANCEL:
1307                removeMessages();
1308                dismissPopupKeyboard();
1309                mAbortKey = true;
1310                showPreview(NOT_A_KEY);
1311                invalidateKey(mCurrentKey);
1312                break;
1313        }
1314        mLastX = touchX;
1315        mLastY = touchY;
1316        return true;
1317    }

1396    private static class SwipeTracker {
1397
1398        static final int NUM_PAST = 4;
1399        static final int LONGEST_PAST_TIME = 200;
1400
1401        final float mPastX[] = new float[NUM_PAST];
1402        final float mPastY[] = new float[NUM_PAST];
1403        final long mPastTime[] = new long[NUM_PAST];
1404
1405        float mYVelocity;
1406        float mXVelocity;
1407
1408        public void clear() {
1409            mPastTime[0] = 0;
1410        }
1411
1412        public void addMovement(MotionEvent ev) {
1413            long time = ev.getEventTime();
1414            final int N = ev.getHistorySize();
1415            for (int i=0; i<N; i++) {
1416                addPoint(ev.getHistoricalX(i), ev.getHistoricalY(i),
1417                        ev.getHistoricalEventTime(i));
1418            }
1419            addPoint(ev.getX(), ev.getY(), time);
1420        }
1421
1422        private void addPoint(float x, float y, long time) {
1423            int drop = -1;
1424            int i;
1425            final long[] pastTime = mPastTime;
1426            for (i=0; i<NUM_PAST; i++) {
1427                if (pastTime[i] == 0) {
1428                    break;
1429                } else if (pastTime[i] < time-LONGEST_PAST_TIME) {
1430                    drop = i;
1431                }
1432            }
1433            if (i == NUM_PAST && drop < 0) {
1434                drop = 0;
1435            }
1436            if (drop == i) drop--;
1437            final float[] pastX = mPastX;
1438            final float[] pastY = mPastY;
1439            if (drop >= 0) {
1440                final int start = drop+1;
1441                final int count = NUM_PAST-drop-1;
1442                System.arraycopy(pastX, start, pastX, 0, count);
1443                System.arraycopy(pastY, start, pastY, 0, count);
1444                System.arraycopy(pastTime, start, pastTime, 0, count);
1445                i -= (drop+1);
1446            }
1447            pastX[i] = x;
1448            pastY[i] = y;
1449            pastTime[i] = time;
1450            i++;
1451            if (i < NUM_PAST) {
1452                pastTime[i] = 0;
1453            }
1454        }
1455
1456        public void computeCurrentVelocity(int units) {
1457            computeCurrentVelocity(units, Float.MAX_VALUE);
1458        }
1459
1460        public void computeCurrentVelocity(int units, float maxVelocity) {
1461            final float[] pastX = mPastX;
1462            final float[] pastY = mPastY;
1463            final long[] pastTime = mPastTime;
1464
1465            final float oldestX = pastX[0];
1466            final float oldestY = pastY[0];
1467            final long oldestTime = pastTime[0];
1468            float accumX = 0;
1469            float accumY = 0;
1470            int N=0;
1471            while (N < NUM_PAST) {
1472                if (pastTime[N] == 0) {
1473                    break;
1474                }
1475                N++;
1476            }
1477
1478            for (int i=1; i < N; i++) {
1479                final int dur = (int)(pastTime[i] - oldestTime);
1480                if (dur == 0) continue;
1481                float dist = pastX[i] - oldestX;
1482                float vel = (dist/dur) * units;   // pixels/frame.
1483                if (accumX == 0) accumX = vel;
1484                else accumX = (accumX + vel) * .5f;
1485
1486                dist = pastY[i] - oldestY;
1487                vel = (dist/dur) * units;   // pixels/frame.
1488                if (accumY == 0) accumY = vel;
1489                else accumY = (accumY + vel) * .5f;
1490            }
1491            mXVelocity = accumX < 0.0f ? Math.max(accumX, -maxVelocity)
1492                    : Math.min(accumX, maxVelocity);
1493            mYVelocity = accumY < 0.0f ? Math.max(accumY, -maxVelocity)
1494                    : Math.min(accumY, maxVelocity);
1495        }
1496
1497        public float getXVelocity() {
1498            return mXVelocity;
1499        }
1500
1501        public float getYVelocity() {
1502            return mYVelocity;
1503        }
1504    }
